23ErrorCode SmRegistry::register_sm(std::unique_ptr<ServiceModel> sm) {
24 if (!sm) {
25 return ErrorCode::INVALID_PARAM;
26 }
27
28 std::lock_guard lock(mutex_);
29
30 uint32_t ran_func = sm->ran_function_id();
31
32 // Check for conflict
33 if (sms_.count(ran_func) > 0) {
34 E3_LOG_ERROR(LOG_TAG) << "SM already registered for RAN function " << ran_func;
35 return ErrorCode::SM_ALREADY_REGISTERED;
36 }
37
38 // Initialize the SM
39 ErrorCode init_result = sm->init();
40 if (init_result != ErrorCode::SUCCESS) {
41 E3_LOG_ERROR(LOG_TAG) << "Failed to initialize SM: " << sm->name();
42 return init_result;
43 }
44
45 E3_LOG_INFO(LOG_TAG) << "Registered SM '" << sm->name()
46 << "' for RAN function " << ran_func;
47
48 sms_[ran_func] = std::move(sm);
49
50 return ErrorCode::SUCCESS;
51}

72ErrorCode SmRegistry::unregister_sm(uint32_t ran_function_id) {
73 std::lock_guard lock(mutex_);
74
75 auto it = sms_.find(ran_function_id);
76 if (it != sms_.end()) {
77 if (it->second->is_running()) {
78 it->second->stop();
79 }
80 it->second->destroy();
81 sms_.erase(it);
82 E3_LOG_INFO(LOG_TAG) << "Unregistered SM for RAN function " << ran_function_id;
83 return ErrorCode::SUCCESS;
84 }
85
86 auto factory_it = factories_.find(ran_function_id);
87 if (factory_it != factories_.end()) {
88 factories_.erase(factory_it);
89 return ErrorCode::SUCCESS;
90 }
91
92 return ErrorCode::SM_NOT_FOUND;
93}

95ServiceModel* SmRegistry::get_by_ran_function(uint32_t ran_function_id) {
96 std::lock_guard lock(mutex_);
97
98 // Direct lookup
99 auto it = sms_.find(ran_function_id);
100 if (it != sms_.end()) {
101 return it->second.get();
102 }
103
104 // Check if there's a factory for this RAN function
105 auto factory_it = factories_.find(ran_function_id);
106 if (factory_it != factories_.end()) {
107 // Create SM from factory
108 auto sm = factory_it->second();
109 if (sm) {
110 ErrorCode init_result = sm->init();
111 if (init_result == ErrorCode::SUCCESS) {
112 ServiceModel* ptr = sm.get();
113 sms_[ran_function_id] = std::move(sm);
114 factories_.erase(factory_it);
115 return ptr;
116 }
117 }
118 }
119
120 return nullptr;
121}

123std::vector<uint32_t> SmRegistry::get_available_ran_functions() const {
124 std::lock_guard lock(mutex_);
125
126 std::vector<uint32_t> result;
127 result.reserve(sms_.size() + factories_.size());

141ErrorCode SmRegistry::start_sm(uint32_t ran_function_id) {
142 std::lock_guard lock(mutex_);
143
144 auto it = sms_.find(ran_function_id);
145 if (it == sms_.end()) {
146 E3_LOG_ERROR(LOG_TAG) << "SM not found for RAN function " << ran_function_id;
147 E3_LOG_ERROR(LOG_TAG) << "SM not found for RAN function " << ran_function_id;
148 return ErrorCode::SM_NOT_FOUND;
149 }
150
151 auto& sm = it->second;
152
153 if (sm->is_running()) {
154 E3_LOG_DEBUG(LOG_TAG) << "SM already running for RAN function " << ran_function_id;
155 return ErrorCode::SUCCESS;
156 }
157
158 ErrorCode result = sm->start();
159 if (result != ErrorCode::SUCCESS) {
160 E3_LOG_ERROR(LOG_TAG) << "Failed to start SM for RAN function " << ran_function_id;
161 return result;
162 }
163
164 E3_LOG_INFO(LOG_TAG) << "Started SM for RAN function " << ran_function_id;
165 return ErrorCode::SUCCESS;
166}

168ErrorCode SmRegistry::stop_sm(uint32_t ran_function_id) {
169 std::lock_guard lock(mutex_);
170
171 auto it = sms_.find(ran_function_id);
172 if (it == sms_.end()) {
173 return ErrorCode::SM_NOT_FOUND;
174 }
175
176 auto& sm = it->second;
177
178 if (!sm->is_running()) {
179 return ErrorCode::SUCCESS;
180 }
181
182 sm->stop();
183 E3_LOG_INFO(LOG_TAG) << "Stopped SM for RAN function " << ran_function_id;
184 return ErrorCode::SUCCESS;
185}

198void SmRegistry::clear() {

200
