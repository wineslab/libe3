1/**
2 * @file dapp_subscription_state.hpp
3 * @brief Per-instance state for the dApp role.
4 *
5 * Holds the dApp-assigned identifier (received from the RAN in the
6 * SetupResponse), the list of RAN functions the RAN advertised, and the
7 * subscriptions this dApp has actively negotiated.
8 *
9 * A dApp may hold MULTIPLE concurrent subscriptions — one per distinct
10 * RAN function. At most one subscription per (dapp, ran_function) pair,
11 * mirroring the RAN-side uniqueness in SubscriptionManager.
12 *
13 * This struct is held by E3Interface when config.role == E3Role::DAPP.
14 * It is the dApp-role counterpart of SubscriptionManager, which stays
15 * strictly RAN-side. Exactly one of `subscription_manager_` / `dapp_state_`
16 * is non-null on a given E3Interface instance.
17 *
18 * SPDX-License-Identifier: Apache-2.0
19 */

41 /**
42 * @brief Store the assignment from a SetupResponse.
43 * Called once after a positive SetupResponse arrives.
44 */
45 void record_setup_response(uint32_t dapp_id,
46 std::vector<RanFunctionDef> functions) {

48 assigned_dapp_id = dapp_id;
49 remote_ran_functions = std::move(functions);

52 /**
53 * @brief Record a successful SubscriptionResponse.
54 * Idempotent: re-recording the same (rf_id, sub_id) is fine.
55 */
56 void record_subscription(uint32_t ran_function_id, uint32_t subscription_id) {
57 std::lock_guard<std::mutex> lock(mu_);
58 ran_function_to_subscription_id[ran_function_id] = subscription_id;
59 subscription_id_to_ran_function[subscription_id] = ran_function_id;
60 }
61
62 /**
63 * @brief Remove the subscription for a given RAN function.
64 * @return true if a subscription was removed; false if none existed.
65 */
66 bool remove_subscription_by_rf(uint32_t ran_function_id) {
67 std::lock_guard<std::mutex> lock(mu_);
68 auto it = ran_function_to_subscription_id.find(ran_function_id);
69 if (it == ran_function_to_subscription_id.end()) {
70 return false;
71 }
72 const uint32_t sub_id = it->second;
73 ran_function_to_subscription_id.erase(it);
74 subscription_id_to_ran_function.erase(sub_id);
75 return true;
76 }
77
78 /**
79 * @brief Remove the subscription with a given subscription ID.
80 * @return true if a subscription was removed; false if none existed.
81 */
82 bool remove_subscription_by_id(uint32_t subscription_id) {
83 std::lock_guard<std::mutex> lock(mu_);
84 auto it = subscription_id_to_ran_function.find(subscription_id);
85 if (it == subscription_id_to_ran_function.end()) {
86 return false;
87 }
88 const uint32_t rf_id = it->second;
89 subscription_id_to_ran_function.erase(it);
90 ran_function_to_subscription_id.erase(rf_id);
91 return true;
92 }

94 /**
95 * @brief Snapshot of active subscription IDs.
96 */
97 std::vector<uint32_t> active_subscriptions() const {
98 std::lock_guard<std::mutex> lock(mu_);
99 std::vector<uint32_t> out;
100 out.reserve(subscription_id_to_ran_function.size());
101 for (const auto& [sub_id, rf_id] : subscription_id_to_ran_function) {
102 (void)rf_id;
103 out.push_back(sub_id);
104 }
105 return out;
106 }

122 /**
123 * @brief Look up subscription ID for a RAN function.
124 */
125 std::optional<uint32_t> subscription_id_for(uint32_t ran_function_id) const {
126 std::lock_guard<std::mutex> lock(mu_);
127 auto it = ran_function_to_subscription_id.find(ran_function_id);
128 if (it == ran_function_to_subscription_id.end()) return std::nullopt;
129 return it->second;
130 }

132 // -----------------------------------------------------------------------
133 // Public state members (kept public for direct inspection in tests and
134 // for handlers that need to update multiple fields atomically inside
135 // E3Interface). Callers MUST hold `mu` when reading or writing these
136 // directly; the helpers above already take the lock.
137 // -----------------------------------------------------------------------
138
139 mutable std::mutex mu;
140
141 /// Assigned by the RAN in SetupResponse. nullopt until setup completes.
142 std::optional<uint32_t> assigned_dapp_id;
143
144 /// RAN functions advertised by the RAN in SetupResponse.
145 std::vector<RanFunctionDef> remote_ran_functions;
146
147 /// rf_id -> sub_id
148 std::unordered_map<uint32_t, uint32_t> ran_function_to_subscription_id;
149 /// sub_id -> rf_id (reverse, used for SubscriptionDelete)
150 std::unordered_map<uint32_t, uint32_t> subscription_id_to_ran_function;
151
152private:
153 // Single mutex; the public-facing `mu` (above) is an alias for
154 // documentation and direct lock_guard use from E3Interface. We keep
155 // a single instance so this->mu_ and the public this->mu refer to the
156 // same lock. Internal helpers use mu_.
157 std::mutex& mu_ = mu;
158};
