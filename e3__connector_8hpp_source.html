21/**
22 * @brief Abstract base class for E3AP transport connectors
23 *
24 * This class defines the transport-agnostic interface that all connectors
25 * must implement. The design mirrors the original C implementation's
26 * function pointer approach, but uses virtual methods for type safety.
27 *
28 * Derived classes:
29 * - PosixE3Connector: POSIX socket-based transport (TCP, SCTP, IPC)
30 * - ZmqE3Connector: ZeroMQ-based transport
31 */
32class E3Connector {
33public:
34 virtual ~E3Connector() = default;

42 /**
43 * @brief Set up the initial connection (setup channel)
44 *
45 * This establishes the REQ/REP channel used for E3 Setup procedures.
46 * @return ErrorCode::SUCCESS on success, error code otherwise

48 virtual ErrorCode setup_initial_connection() = 0;

50 /**

72 virtual ErrorCode setup_inbound_connection() = 0;
