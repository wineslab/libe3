62ErrorCode ZmqE3Connector::setup_initial_connection() {
63 // Create ZMQ context
64 context_ = zmq_ctx_new();
65 if (!context_) {
66 E3_LOG_ERROR(LOG_TAG) << "Failed to create ZMQ context";
67 return ErrorCode::CONNECTION_FAILED;
68 }
69
70 // Set I/O threads
71 if (zmq_ctx_set(context_, ZMQ_IO_THREADS, static_cast<int>(io_threads_)) != 0) {
72 E3_LOG_ERROR(LOG_TAG) << "Failed to set ZMQ I/O threads: " << zmq_strerror(errno);
73 zmq_ctx_destroy(context_);
74 context_ = nullptr;
75 return ErrorCode::CONNECTION_FAILED;
76 }
77
78 // Create IPC directory if using IPC transport
79 if (transport_layer_ == E3TransportLayer::IPC) {
80 struct stat st{};
81 if (stat(IPC_BASE_DIR, &st) == -1) {
82 if (mkdir(IPC_BASE_DIR, 0777) == -1) {
83 E3_LOG_ERROR(LOG_TAG) << "Failed to create IPC directory: " << strerror(errno);
84 return ErrorCode::CONNECTION_FAILED;
85 }
86 E3_LOG_INFO(LOG_TAG) << "Created IPC directory: " << IPC_BASE_DIR;
87 }
88
89 // Set directory permissions
90 if (chmod(IPC_BASE_DIR, 0777) != 0) {
91 E3_LOG_WARN(LOG_TAG) << "Failed to set IPC directory permissions";
92 }
93
94 // Try to change group to "dapp"
95 struct group* grp = getgrnam("dapp");
96 if (grp) {
97 if (chown(IPC_BASE_DIR, static_cast<uid_t>(-1), grp->gr_gid) == -1) {
98 E3_LOG_WARN(LOG_TAG) << "Failed to change IPC directory group";
99 }
100 }
101 }
102
103 // Create REP socket for setup
104 setup_socket_ = zmq_socket(context_, ZMQ_REP);
105 if (!setup_socket_) {
106 E3_LOG_ERROR(LOG_TAG) << "Failed to create setup socket: " << zmq_strerror(errno);
107 return ErrorCode::CONNECTION_FAILED;
108 }
109
110 // Set receive timeout to allow graceful shutdown
111 int recv_timeout = RECV_TIMEOUT_MS;
112 zmq_setsockopt(setup_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
113
114 int ret = zmq_bind(setup_socket_, setup_endpoint_.c_str());
115 if (ret != 0) {
116 E3_LOG_ERROR(LOG_TAG) << "Failed to bind setup socket: " << zmq_strerror(errno);
117 return ErrorCode::CONNECTION_FAILED;
118 }
119
120 // Set IPC permissions if needed
121 if (transport_layer_ == E3TransportLayer::IPC) {
122 setup_ipc_permissions(setup_endpoint_);
123 }
124
125 E3_LOG_INFO(LOG_TAG) << "Setup socket bound to " << setup_endpoint_;
126 connected_ = true;
127
128 return ErrorCode::SUCCESS;
129}

131int ZmqE3Connector::recv_setup_request(std::vector<uint8_t>& buffer) {
132 if (!setup_socket_) {
133 return static_cast<int>(ErrorCode::NOT_CONNECTED);
134 }
135
136 buffer.resize(DEFAULT_BUFFER_SIZE);
137 int ret = zmq_recv(setup_socket_, buffer.data(), buffer.size(), 0);
138 if (ret < 0) {
139 if (errno == EAGAIN) {
140 // Timeout - return 0 to indicate no data (allows shutdown check)
141 return 0;
142 }
143 E3_LOG_ERROR(LOG_TAG) << "Failed to receive setup request: " << zmq_strerror(errno);
144 reset_setup_socket();
145 return static_cast<int>(ErrorCode::TRANSPORT_ERROR);
146 }
147
148 buffer.resize(static_cast<size_t>(ret));
149 E3_LOG_DEBUG(LOG_TAG) << "Received setup request: " << ret << " bytes";
150 return ret;
151}

169ErrorCode ZmqE3Connector::setup_inbound_connection() {
170 if (!context_) {
171 return ErrorCode::NOT_INITIALIZED;
172 }
173
174 inbound_socket_ = zmq_socket(context_, ZMQ_SUB);
175 if (!inbound_socket_) {
176 E3_LOG_ERROR(LOG_TAG) << "Failed to create inbound socket: " << zmq_strerror(errno);
177 return ErrorCode::CONNECTION_FAILED;
178 }
179
180 // Subscribe to all messages
181 zmq_setsockopt(inbound_socket_, ZMQ_SUBSCRIBE, "", 0);
182
183 // Keep only the latest message (conflate)
184 int conflate = 1;
185 zmq_setsockopt(inbound_socket_, ZMQ_CONFLATE, &conflate, sizeof(conflate));
186
187 // Set receive timeout to allow graceful shutdown
188 int recv_timeout = RECV_TIMEOUT_MS;
189 zmq_setsockopt(inbound_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
190
191 int ret = zmq_bind(inbound_socket_, inbound_endpoint_.c_str());
192 if (ret != 0) {
193 E3_LOG_ERROR(LOG_TAG) << "Failed to bind inbound socket: " << zmq_strerror(errno);
194 return ErrorCode::CONNECTION_FAILED;
195 }
196
197 if (transport_layer_ == E3TransportLayer::IPC) {
198 setup_ipc_permissions(inbound_endpoint_);
199 }
200
201 E3_LOG_INFO(LOG_TAG) << "Inbound socket bound to " << inbound_endpoint_;
202 return ErrorCode::SUCCESS;
203}

266void ZmqE3Connector::dispose() {
267 E3_LOG_DEBUG(LOG_TAG) << "Disposing ZMQ connector";
268
269 if (setup_socket_) {
270 zmq_close(setup_socket_);
271 setup_socket_ = nullptr;
272 }
273
274 if (inbound_socket_) {
275 zmq_close(inbound_socket_);
276 inbound_socket_ = nullptr;
277 }
278
279 if (outbound_socket_) {
280 zmq_close(outbound_socket_);
281 outbound_socket_ = nullptr;
282 }
283
284 if (context_) {
285 zmq_ctx_destroy(context_);
286 context_ = nullptr;
287 }
288
289 // Clean up IPC files if using IPC transport
290 if (transport_layer_ == E3TransportLayer::IPC) {
291 // Extract file paths from IPC endpoints (ipc:///path)
292 auto extract_path = [](const std::string& endpoint) -> std::string {
293 const std::string prefix = "ipc://";
294 if (endpoint.compare(0, prefix.length(), prefix) == 0) {
295 return endpoint.substr(prefix.length());
296 }
297 return "";
298 };
299
300 std::string path;
301 path = extract_path(setup_endpoint_);
302 if (!path.empty()) unlink(path.c_str());
303
304 path = extract_path(inbound_endpoint_);
305 if (!path.empty()) unlink(path.c_str());
306
307 path = extract_path(outbound_endpoint_);
308 if (!path.empty()) unlink(path.c_str());
309
310 rmdir(IPC_BASE_DIR);
311 }
312
313 connected_ = false;
314 E3_LOG_INFO(LOG_TAG) << "ZMQ connector disposed";
315}

324void ZmqE3Connector::setup_ipc_permissions(const std::string& endpoint) {
325 // Extract file path from IPC endpoint (ipc:///path)
326 const std::string prefix = "ipc://";
327 if (endpoint.compare(0, prefix.length(), prefix) != 0) {
328 return;
329 }
330
331 std::string path = endpoint.substr(prefix.length());
332
333 if (chmod(path.c_str(), 0666) == -1) {
334 E3_LOG_WARN(LOG_TAG) << "Failed to set permissions on " << path << ": " << strerror(errno);
335 } else {
336 E3_LOG_DEBUG(LOG_TAG) << "Set permissions on " << path;
337 }

340bool ZmqE3Connector::reset_setup_socket() {
341 E3_LOG_WARN(LOG_TAG) << "Resetting setup socket after error";
342
343 if (setup_socket_) {
344 zmq_close(setup_socket_);
345 setup_socket_ = nullptr;
346 }
347
348 if (!context_) {
349 E3_LOG_ERROR(LOG_TAG) << "Cannot reset setup socket without context";
350 return false;
351 }
352
353 setup_socket_ = zmq_socket(context_, ZMQ_REP);
354 if (!setup_socket_) {
355 E3_LOG_ERROR(LOG_TAG) << "Failed to recreate setup socket: " << zmq_strerror(errno);
356 return false;
357 }
358
359 int recv_timeout = RECV_TIMEOUT_MS;
360 zmq_setsockopt(setup_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
361
362 int ret = zmq_bind(setup_socket_, setup_endpoint_.c_str());
363 if (ret != 0) {
364 E3_LOG_ERROR(LOG_TAG) << "Failed to rebind setup socket: " << zmq_strerror(errno);
365 zmq_close(setup_socket_);
366 setup_socket_ = nullptr;
367 return false;
368 }
369
370 if (transport_layer_ == E3TransportLayer::IPC) {
371 setup_ipc_permissions(setup_endpoint_);
372 }
373
374 E3_LOG_INFO(LOG_TAG) << "Setup socket reset and bound to " << setup_endpoint_;
375 return true;
376}
