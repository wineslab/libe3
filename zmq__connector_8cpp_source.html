29ZmqE3Connector::ZmqE3Connector(
30 E3TransportLayer transport_layer,
31 const std::string& setup_endpoint,
32 const std::string& inbound_endpoint,
33 const std::string& outbound_endpoint,
34 uint16_t setup_port,
35 uint16_t inbound_port,
36 uint16_t outbound_port,
37 size_t io_threads
38)
39 : transport_layer_(transport_layer)
40 , io_threads_(io_threads)
41 , setup_port_(setup_port)
42 , inbound_port_(inbound_port)
43 , outbound_port_(outbound_port)
44{
45 // For TCP, derive the endpoint strings from the port numbers. The
46 // bind/connect-direction adjustment is handled at the call sites below
47 // (RAN binds tcp://*:N; dApp connects to tcp://127.0.0.1:N). For IPC
48 // both sides use the same ipc:// string.
49 if (transport_layer == E3TransportLayer::TCP) {
50 setup_endpoint_ = "tcp://*:" + std::to_string(setup_port);
51 inbound_endpoint_ = "tcp://*:" + std::to_string(inbound_port);
52 outbound_endpoint_ = "tcp://*:" + std::to_string(outbound_port);
53 } else {
54 setup_endpoint_ = setup_endpoint;
55 inbound_endpoint_ = inbound_endpoint;
56 outbound_endpoint_ = outbound_endpoint;
57 }
58
59 E3_LOG_INFO(LOG_TAG) << "Creating ZMQ connector";
60 E3_LOG_DEBUG(LOG_TAG) << " Setup endpoint: " << setup_endpoint_;
61 E3_LOG_DEBUG(LOG_TAG) << " Inbound endpoint: " << inbound_endpoint_;
62 E3_LOG_DEBUG(LOG_TAG) << " Outbound endpoint: " << outbound_endpoint_;
63}

69ErrorCode ZmqE3Connector::setup_initial_connection() {
70 // Create ZMQ context
71 context_ = zmq_ctx_new();
72 if (!context_) {
73 E3_LOG_ERROR(LOG_TAG) << "Failed to create ZMQ context";
74 return ErrorCode::CONNECTION_FAILED;
75 }
76
77 // Set I/O threads
78 if (zmq_ctx_set(context_, ZMQ_IO_THREADS, static_cast<int>(io_threads_)) != 0) {
79 E3_LOG_ERROR(LOG_TAG) << "Failed to set ZMQ I/O threads: " << zmq_strerror(errno);
80 zmq_ctx_destroy(context_);
81 context_ = nullptr;
82 return ErrorCode::CONNECTION_FAILED;
83 }
84
85 // Create IPC directory if using IPC transport
86 if (transport_layer_ == E3TransportLayer::IPC) {
87 struct stat st{};
88 if (stat(IPC_BASE_DIR, &st) == -1) {
89 if (mkdir(IPC_BASE_DIR, 0777) == -1) {
90 E3_LOG_ERROR(LOG_TAG) << "Failed to create IPC directory: " << strerror(errno);
91 return ErrorCode::CONNECTION_FAILED;
92 }
93 E3_LOG_INFO(LOG_TAG) << "Created IPC directory: " << IPC_BASE_DIR;
94 }
95
96 // Set directory permissions
97 if (chmod(IPC_BASE_DIR, 0777) != 0) {
98 E3_LOG_WARN(LOG_TAG) << "Failed to set IPC directory permissions";
99 }
100
101 // Try to change group to "dapp"
102 struct group* grp = getgrnam("dapp");
103 if (grp) {
104 if (chown(IPC_BASE_DIR, static_cast<uid_t>(-1), grp->gr_gid) == -1) {
105 E3_LOG_WARN(LOG_TAG) << "Failed to change IPC directory group";
106 }
107 }
108 }
109
110 // Create REP socket for setup
111 setup_socket_ = zmq_socket(context_, ZMQ_REP);
112 if (!setup_socket_) {
113 E3_LOG_ERROR(LOG_TAG) << "Failed to create setup socket: " << zmq_strerror(errno);
114 return ErrorCode::CONNECTION_FAILED;
115 }
116
117 // Set receive timeout to allow graceful shutdown
118 int recv_timeout = RECV_TIMEOUT_MS;
119 zmq_setsockopt(setup_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
120
121 int ret = zmq_bind(setup_socket_, setup_endpoint_.c_str());
122 if (ret != 0) {
123 E3_LOG_ERROR(LOG_TAG) << "Failed to bind setup socket: " << zmq_strerror(errno);
124 return ErrorCode::CONNECTION_FAILED;
125 }
126
127 // Set IPC permissions if needed
128 if (transport_layer_ == E3TransportLayer::IPC) {
129 setup_ipc_permissions(setup_endpoint_);
130 }
131
132 E3_LOG_INFO(LOG_TAG) << "Setup socket bound to " << setup_endpoint_;
133 connected_ = true;
134
135 return ErrorCode::SUCCESS;
136}

138int ZmqE3Connector::recv_setup_request(std::vector<uint8_t>& buffer) {
139 if (!setup_socket_) {
140 return static_cast<int>(ErrorCode::NOT_CONNECTED);
141 }
142
143 buffer.resize(DEFAULT_BUFFER_SIZE);
144 int ret = zmq_recv(setup_socket_, buffer.data(), buffer.size(), 0);
145 if (ret < 0) {
146 if (errno == EAGAIN) {
147 // Timeout - return 0 to indicate no data (allows shutdown check)
148 return 0;
149 }
150 E3_LOG_ERROR(LOG_TAG) << "Failed to receive setup request: " << zmq_strerror(errno);
151 reset_setup_socket();
152 return static_cast<int>(ErrorCode::TRANSPORT_ERROR);
153 }
154
155 buffer.resize(static_cast<size_t>(ret));
156 E3_LOG_DEBUG(LOG_TAG) << "Received setup request: " << ret << " bytes";
157 return ret;
158}

176ErrorCode ZmqE3Connector::setup_inbound_connection() {
177 if (!context_) {
178 return ErrorCode::NOT_INITIALIZED;
179 }
180
181 inbound_socket_ = zmq_socket(context_, ZMQ_SUB);
182 if (!inbound_socket_) {
183 E3_LOG_ERROR(LOG_TAG) << "Failed to create inbound socket: " << zmq_strerror(errno);
184 return ErrorCode::CONNECTION_FAILED;
185 }
186
187 // Subscribe to all messages
188 zmq_setsockopt(inbound_socket_, ZMQ_SUBSCRIBE, "", 0);
189
190 // No ZMQ_CONFLATE: multiple dApps multiplex onto this SUB; conflating would
191 // drop all but the newest message and starve all but one dApp (issue #15).
192
193 // Set receive timeout to allow graceful shutdown
194 int recv_timeout = RECV_TIMEOUT_MS;
195 zmq_setsockopt(inbound_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
196
197 int ret = zmq_bind(inbound_socket_, inbound_endpoint_.c_str());
198 if (ret != 0) {
199 E3_LOG_ERROR(LOG_TAG) << "Failed to bind inbound socket: " << zmq_strerror(errno);
200 return ErrorCode::CONNECTION_FAILED;
201 }
202
203 if (transport_layer_ == E3TransportLayer::IPC) {
204 setup_ipc_permissions(inbound_endpoint_);
205 }
206
207 E3_LOG_INFO(LOG_TAG) << "Inbound socket bound to " << inbound_endpoint_;
208 return ErrorCode::SUCCESS;

273// Convert a bind-style endpoint ("tcp://*:9990") to a connect-style one
274// ("tcp://127.0.0.1:9990"). IPC endpoints are returned unchanged.
275std::string to_connect_endpoint(const std::string& bind_ep) {
276 const std::string prefix = "tcp://*:";
277 if (bind_ep.compare(0, prefix.size(), prefix) == 0) {
278 return "tcp://127.0.0.1:" + bind_ep.substr(prefix.size());
279 }
280 return bind_ep;
281}

284// ===========================================================================
285// Client-side (dApp role) implementations
286//
287// The dApp ROLE inverts the bind/connect direction on all three channels:
288// - setup_endpoint : REQ connect (RAN binds REP)
289// - publisher_endpoint: SUB connect (RAN binds PUB) — this is our INBOUND
290// - subscriber_endpoint: PUB connect (RAN binds SUB) — this is our OUTBOUND
291// Socket TYPES on the inbound/outbound pair are unchanged; only the
292// bind/connect direction flips and the role of the two endpoints swaps.
293// ===========================================================================
294
295ErrorCode ZmqE3Connector::setup_initial_connection_client() {
296 // Create ZMQ context (same as RAN-side init)
297 context_ = zmq_ctx_new();
298 if (!context_) {
299 E3_LOG_ERROR(LOG_TAG) << "Failed to create ZMQ context";
300 return ErrorCode::CONNECTION_FAILED;
301 }
302 if (zmq_ctx_set(context_, ZMQ_IO_THREADS, static_cast<int>(io_threads_)) != 0) {
303 E3_LOG_ERROR(LOG_TAG) << "Failed to set ZMQ I/O threads: " << zmq_strerror(errno);
304 zmq_ctx_destroy(context_);
305 context_ = nullptr;
306 return ErrorCode::CONNECTION_FAILED;
307 }
308
309 // REQ socket — the dApp INITIATES setup
310 setup_socket_ = zmq_socket(context_, ZMQ_REQ);
311 if (!setup_socket_) {
312 E3_LOG_ERROR(LOG_TAG) << "Failed to create REQ setup socket: " << zmq_strerror(errno);
313 return ErrorCode::CONNECTION_FAILED;
314 }
315
316 int recv_timeout = RECV_TIMEOUT_MS;
317 zmq_setsockopt(setup_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
318 int linger = 0; // Don't block on close if RAN never replied
319 zmq_setsockopt(setup_socket_, ZMQ_LINGER, &linger, sizeof(linger));
320
321 const std::string setup_cep = to_connect_endpoint(setup_endpoint_);
322 int ret = zmq_connect(setup_socket_, setup_cep.c_str());
323 if (ret != 0) {
324 E3_LOG_ERROR(LOG_TAG) << "Failed to connect setup socket: " << zmq_strerror(errno);
325 return ErrorCode::CONNECTION_FAILED;
326 }
327 E3_LOG_INFO(LOG_TAG) << "Setup REQ socket connected to " << setup_cep;
328 connected_ = true;
329 return ErrorCode::SUCCESS;
330}

357ErrorCode ZmqE3Connector::setup_inbound_connection_client() {
358 if (!context_) return ErrorCode::NOT_INITIALIZED;
359
360 // dApp INBOUND = RAN's publisher_endpoint (where RAN PUB-binds)
361 inbound_socket_ = zmq_socket(context_, ZMQ_SUB);
362 if (!inbound_socket_) {
363 E3_LOG_ERROR(LOG_TAG) << "Failed to create inbound SUB socket: " << zmq_strerror(errno);
364 return ErrorCode::CONNECTION_FAILED;
365 }
366 zmq_setsockopt(inbound_socket_, ZMQ_SUBSCRIBE, "", 0);
367 // No ZMQ_CONFLATE: the RAN broadcasts per-dApp indications over one PUB and
368 // each dApp filters its own; conflating would starve all but one (issue #15).
369 int recv_timeout = RECV_TIMEOUT_MS;
370 zmq_setsockopt(inbound_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
371
372 const std::string inbound_cep = to_connect_endpoint(outbound_endpoint_);
373 int ret = zmq_connect(inbound_socket_, inbound_cep.c_str());
374 if (ret != 0) {
375 E3_LOG_ERROR(LOG_TAG) << "Failed to connect inbound SUB to "
376 << inbound_cep << ": " << zmq_strerror(errno);
377 return ErrorCode::CONNECTION_FAILED;
378 }
379 E3_LOG_INFO(LOG_TAG) << "Inbound SUB socket connected to " << inbound_cep;
380 return ErrorCode::SUCCESS;
381}

383ErrorCode ZmqE3Connector::setup_outbound_connection_client() {
384 if (!context_) return ErrorCode::NOT_INITIALIZED;
385
386 // dApp OUTBOUND = RAN's subscriber_endpoint (where RAN SUB-binds)
387 outbound_socket_ = zmq_socket(context_, ZMQ_PUB);
388 if (!outbound_socket_) {
389 E3_LOG_ERROR(LOG_TAG) << "Failed to create outbound PUB socket: " << zmq_strerror(errno);
390 return ErrorCode::CONNECTION_FAILED;
391 }
392
393 const std::string outbound_cep = to_connect_endpoint(inbound_endpoint_);
394 int ret = zmq_connect(outbound_socket_, outbound_cep.c_str());
395 if (ret != 0) {
396 E3_LOG_ERROR(LOG_TAG) << "Failed to connect outbound PUB to "
397 << outbound_cep << ": " << zmq_strerror(errno);
398 return ErrorCode::CONNECTION_FAILED;
399 }
400 E3_LOG_INFO(LOG_TAG) << "Outbound PUB socket connected to " << outbound_cep;
401
402 // ZMQ PUB/SUB slow-joiner: a freshly connected PUB silently drops outgoing
403 // messages until the peer SUB's subscription has propagated to it. The
404 // dApp's first outbound message is the subscription request; without a
405 // brief settle it can be lost — reliably so for a second peer over TCP,
406 // where connection establishment is slower. Wait for the link to settle.
407 std::this_thread::sleep_for(std::chrono::milliseconds(PUB_CONNECT_SETTLE_MS));
408 return ErrorCode::SUCCESS;
409}

411void ZmqE3Connector::dispose() {
412 E3_LOG_DEBUG(LOG_TAG) << "Disposing ZMQ connector";
413
414 if (setup_socket_) {
415 zmq_close(setup_socket_);
416 setup_socket_ = nullptr;
417 }
418
419 if (inbound_socket_) {
420 zmq_close(inbound_socket_);
421 inbound_socket_ = nullptr;
422 }
423
424 if (outbound_socket_) {
425 zmq_close(outbound_socket_);
426 outbound_socket_ = nullptr;
427 }
428
429 if (context_) {
430 zmq_ctx_destroy(context_);
431 context_ = nullptr;
432 }
433
434 // Clean up IPC files if using IPC transport AND we own them (RAN side
435 // binds, which creates the socket files; dApp side connects, so it has
436 // nothing to unlink).
437 if (transport_layer_ == E3TransportLayer::IPC && role_ == E3Role::RAN) {
438 // Extract file paths from IPC endpoints (ipc:///path)
439 auto extract_path = [](const std::string& endpoint) -> std::string {
440 const std::string prefix = "ipc://";
441 if (endpoint.compare(0, prefix.length(), prefix) == 0) {
442 return endpoint.substr(prefix.length());
443 }
444 return "";
445 };
446
447 std::string path;
448 path = extract_path(setup_endpoint_);
449 if (!path.empty()) unlink(path.c_str());
450
451 path = extract_path(inbound_endpoint_);
452 if (!path.empty()) unlink(path.c_str());
453
454 path = extract_path(outbound_endpoint_);
455 if (!path.empty()) unlink(path.c_str());
456
457 rmdir(IPC_BASE_DIR);
458 }
459
460 connected_ = false;
461 E3_LOG_INFO(LOG_TAG) << "ZMQ connector disposed";
462}

472 // Extract file path from IPC endpoint (ipc:///path)
473 const std::string prefix = "ipc://";
474 if (endpoint.compare(0, prefix.length(), prefix) != 0) {
475 return;

483 E3_LOG_DEBUG(LOG_TAG) << "Set permissions on " << path;

487bool ZmqE3Connector::reset_setup_socket() {
488 E3_LOG_WARN(LOG_TAG) << "Resetting setup socket after error";
489
490 if (setup_socket_) {
491 zmq_close(setup_socket_);
492 setup_socket_ = nullptr;
493 }
494
495 if (!context_) {
496 E3_LOG_ERROR(LOG_TAG) << "Cannot reset setup socket without context";
497 return false;
498 }
499
500 setup_socket_ = zmq_socket(context_, ZMQ_REP);
501 if (!setup_socket_) {
502 E3_LOG_ERROR(LOG_TAG) << "Failed to recreate setup socket: " << zmq_strerror(errno);
503 return false;
504 }
505
506 int recv_timeout = RECV_TIMEOUT_MS;
507 zmq_setsockopt(setup_socket_, ZMQ_RCVTIMEO, &recv_timeout, sizeof(recv_timeout));
508
509 int ret = zmq_bind(setup_socket_, setup_endpoint_.c_str());
510 if (ret != 0) {
511 E3_LOG_ERROR(LOG_TAG) << "Failed to rebind setup socket: " << zmq_strerror(errno);
512 zmq_close(setup_socket_);
513 setup_socket_ = nullptr;
514 return false;
515 }
516
517 if (transport_layer_ == E3TransportLayer::IPC) {
518 setup_ipc_permissions(setup_endpoint_);
519 }
520
521 E3_LOG_INFO(LOG_TAG) << "Setup socket reset and bound to " << setup_endpoint_;
522 return true;
523}
