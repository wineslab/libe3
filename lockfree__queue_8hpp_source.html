2 * @file lockfree_queue.hpp
3 * @brief Lock-free bounded blocking queue for the E3AP hot paths
4 *
5 * Wraps the lock-free MPMC ring buffer (see mpmc_queue.hpp) in a small,
6 * reusable object that adds blocking pop variants and shutdown semantics.
7 * It eliminates mutex contention on the hot publish path and reduces latency
8 * jitter in sub-millisecond control loops.
9 *
10 * libe3 uses two specialisations:
11 * - LockFreeQueue<Pdu> — outbound E3AP PDUs (publisher/outbound thread
12 * consumes; many threads produce).
13 * - LockFreeQueue<DAppReport> — inbound dApp reports handed from the RAN
14 * inbound thread to the report worker thread.

32/**
33 * @brief Lock-free bounded blocking queue.
34 *
35 * Built on an MPMC lock-free ring buffer. Blocking pop variants use an
36 * adaptive spin-wait strategy:
37 * 1. Spin with CPU pause hints (lowest latency, nanoseconds)
38 * 2. Thread yield (cooperative, microseconds)
39 * 3. Short sleep (50 µs) (idle wait, avoids busy-loop when quiet)
40 *
41 * @tparam T Element type. Must be default-constructible and movable.
42 */
43template<typename T>
44class LockFreeQueue {
45public:
46 /**
47 * @brief Construct a LockFreeQueue.
48 * @param capacity Minimum ring buffer capacity (rounded up to next power
49 * of two, default 128).
50 */
51 explicit LockFreeQueue(size_t capacity = 128)
52 : ring_(capacity)
53 {
54 E3_LOG_DEBUG(LOG_TAG) << "Lock-free queue created, capacity="
55 << ring_.capacity();
56 }
57
58 /**
59 * @brief Destructor – signals shutdown so any blocked pop() returns.
60 */
61 ~LockFreeQueue() {
62 shutdown();
63 E3_LOG_DEBUG(LOG_TAG) << "Lock-free queue destroyed";
64 }
65
66 // Non-copyable, non-movable
67 LockFreeQueue(const LockFreeQueue&) = delete;
68 LockFreeQueue& operator=(const LockFreeQueue&) = delete;
69 LockFreeQueue(LockFreeQueue&&) = delete;
70 LockFreeQueue& operator=(LockFreeQueue&&) = delete;
71
72 /**
73 * @brief Enqueue an item (non-blocking, lock-free).
74 * @return ErrorCode::SUCCESS on success.
75 * @return ErrorCode::BUFFER_TOO_SMALL if the ring buffer is full.
76 * @return ErrorCode::NOT_INITIALIZED if shutdown() has been called.
77 */
78 ErrorCode push(T item) {
79 if (shutdown_.load(std::memory_order_relaxed)) {
80 return ErrorCode::NOT_INITIALIZED;
81 }
82
83 if (!ring_.try_push(std::move(item))) {
84 E3_LOG_WARN(LOG_TAG) << "Queue full, dropping message";
85 return ErrorCode::BUFFER_TOO_SMALL;
86 }
87
88 E3_LOG_TRACE(LOG_TAG) << "Pushed item";
89 return ErrorCode::SUCCESS;
90 }
91
92 /**
93 * @brief Dequeue an item, blocking indefinitely until one is available.
94 *
95 * Returns a default-constructed T{} if shutdown() is called while waiting.
96 */
97 T pop() {
98 T item;
99
100 // Phase 1: CPU-pause spin (nanosecond latency when producer is fast)
101 for (size_t i = 0; i < SPIN_COUNT; ++i) {
102 if (ring_.try_pop(item)) return item;
103 if (shutdown_.load(std::memory_order_relaxed)) return T{};
104 cpu_relax();
105 }
106
107 // Phase 2: Cooperative yield (microsecond range)
108 for (size_t i = 0; i < YIELD_COUNT; ++i) {
109 if (ring_.try_pop(item)) return item;
110 if (shutdown_.load(std::memory_order_relaxed)) return T{};
111 std::this_thread::yield();
112 }
113
114 // Phase 3: Short sleep until data arrives or shutdown is signalled
115 while (true) {
116 if (ring_.try_pop(item)) return item;
117 if (shutdown_.load(std::memory_order_relaxed)) return T{};
118 std::this_thread::sleep_for(SLEEP_DURATION);
119 }
120 }
121
122 /**
123 * @brief Dequeue an item with a maximum wait duration.
124 * @return The item on success; std::nullopt on timeout or shutdown.
125 */
126 std::optional<T> pop(std::chrono::milliseconds timeout) {
127 T item;
128 auto deadline = std::chrono::steady_clock::now() + timeout;
129
130 // Phase 1: CPU-pause spin
131 for (size_t i = 0; i < SPIN_COUNT; ++i) {
132 if (ring_.try_pop(item)) return item;
133 if (shutdown_.load(std::memory_order_relaxed)) return std::nullopt;
134 cpu_relax();
135 }
136
137 // Phase 2: Cooperative yield
138 for (size_t i = 0; i < YIELD_COUNT; ++i) {
139 if (ring_.try_pop(item)) return item;
140 if (shutdown_.load(std::memory_order_relaxed)) return std::nullopt;
141 std::this_thread::yield();
142 }
143
144 // Phase 3: Timed sleep
145 while (std::chrono::steady_clock::now() < deadline) {
146 if (ring_.try_pop(item)) return item;
147 if (shutdown_.load(std::memory_order_relaxed)) return std::nullopt;
148 std::this_thread::sleep_for(SLEEP_DURATION);
149 }
150
151 // One last attempt after deadline
152 if (ring_.try_pop(item)) return item;
153 return std::nullopt;
154 }
155
156 /**
157 * @brief Try to dequeue without blocking.
158 * @return The item if one was available; std::nullopt otherwise.
159 */
160 std::optional<T> try_pop() {
161 T item;
162 if (ring_.try_pop(item)) return item;
163 return std::nullopt;
164 }
165
166 /** @brief Return true if the queue appears empty (approximate). */
167 bool empty() const { return ring_.empty_approx(); }
168
169 /** @brief Return the approximate number of items in the queue. */
170 size_t size() const { return ring_.size_approx(); }
171
172 /** @brief Return the ring buffer capacity (always a power of two). */
173 size_t capacity() const noexcept { return ring_.capacity(); }
174
175 /** @brief Discard all items currently in the queue. */
176 void clear() {
177 T item;
178 while (ring_.try_pop(item)) {}
179 E3_LOG_DEBUG(LOG_TAG) << "Queue cleared";
180 }
181
182 /**
183 * @brief Signal shutdown so that all blocked pop() calls return promptly.
184 */
185 void shutdown() {
186 shutdown_.store(true, std::memory_order_relaxed);
187 E3_LOG_DEBUG(LOG_TAG) << "Queue shutdown signalled";
188 }
189
190 /** @brief Return true if shutdown() has been called. */
191 bool is_shutdown() const {
192 return shutdown_.load(std::memory_order_relaxed);
193 }
194
195private:
196 static constexpr const char* LOG_TAG = "Queue";
197
198 MpmcQueue<T> ring_;
199 std::atomic<bool> shutdown_{false};
200
201 // Adaptive spin-wait tuning constants
202 static constexpr size_t SPIN_COUNT = 40; ///< CPU-pause iterations
203 static constexpr size_t YIELD_COUNT = 100; ///< thread-yield iterations
204 /// Sleep duration between attempts in the slow path (µs)
205 static constexpr auto SLEEP_DURATION = std::chrono::microseconds(50);
206};
