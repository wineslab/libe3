1/**
2 * @file response_queue.hpp
3 * @brief Lock-free bounded queue for E3AP outbound PDUs
4 *
5 * Replaces the original mutex + condition-variable implementation with a
6 * lock-free MPMC ring buffer (see mpmc_queue.hpp) to eliminate mutex
7 * contention on the hot publish path and to reduce latency jitter in
8 * sub-millisecond control loops.
9 *
10 * Producer/consumer pattern in libe3:
11 * - Producers: subscriber_loop thread, sm_data_handler_loop thread, and
12 * any caller of E3Interface::queue_outbound() (MPSC / MPMC).
13 * - Consumer: publisher_loop thread (single consumer).
14 *
15 * SPDX-License-Identifier: Apache-2.0
16 */

29/**
30 * @brief Lock-free bounded queue for E3AP PDUs.
31 *
32 * Provides the same API as the previous mutex-based implementation but uses
33 * an MPMC lock-free ring buffer internally. Blocking pop variants use an
34 * adaptive spin-wait strategy:
35 * 1. Spin with CPU pause hints (lowest latency, nanoseconds)
36 * 2. Thread yield (cooperative, microseconds)
37 * 3. Short sleep (50 µs) (idle wait, avoids busy-loop when quiet)
38 */
39class ResponseQueue {
40public:
41 /**
42 * @brief Construct a ResponseQueue.
43 * @param capacity Minimum ring buffer capacity (rounded up to next power
44 * of two, default 128).
45 */
46 explicit ResponseQueue(size_t capacity = 128);
47
48 /**
49 * @brief Destructor – signals shutdown so any blocked pop() returns.
50 */
51 ~ResponseQueue();
52
53 // Non-copyable, non-movable
54 ResponseQueue(const ResponseQueue&) = delete;
55 ResponseQueue& operator=(const ResponseQueue&) = delete;
56 ResponseQueue(ResponseQueue&&) = delete;
57 ResponseQueue& operator=(ResponseQueue&&) = delete;
58
59 /**
60 * @brief Enqueue a PDU (non-blocking, lock-free).
61 * @return ErrorCode::SUCCESS on success.
62 * @return ErrorCode::BUFFER_TOO_SMALL if the ring buffer is full.
63 * @return ErrorCode::NOT_INITIALIZED if shutdown() has been called.
64 */
65 ErrorCode push(Pdu pdu);
66
67 /**
68 * @brief Dequeue a PDU, blocking indefinitely until one is available.
69 *
70 * Returns an empty Pdu{} if shutdown() is called while waiting.
71 */
72 Pdu pop();
73
74 /**
75 * @brief Dequeue a PDU with a maximum wait duration.

78 std::optional<Pdu> pop(std::chrono::milliseconds timeout);

80 /**

84 std::optional<Pdu> try_pop();

86 /** @brief Return true if the queue appears empty (approximate). */
87 bool empty() const;
88
89 /** @brief Return the approximate number of items in the queue. */
90 size_t size() const;

92 /** @brief Return the ring buffer capacity (always a power of two). */
93 size_t capacity() const noexcept { return ring_.capacity(); }
94
95 /** @brief Discard all items currently in the queue. */
96 void clear();
97
98 /**
99 * @brief Signal shutdown so that all blocked pop() calls return promptly.
100 */
101 void shutdown();
102
103 /** @brief Return true if shutdown() has been called. */
104 bool is_shutdown() const;
105
106private:
107 MpmcQueue<Pdu> ring_;
108 std::atomic<bool> shutdown_{false};
109
110 // Adaptive spin-wait tuning constants
111 static constexpr size_t SPIN_COUNT = 40; ///< CPU-pause iterations
112 static constexpr size_t YIELD_COUNT = 100; ///< thread-yield iterations
113 /// Sleep duration between attempts in the slow path (µs)
114 static constexpr auto SLEEP_DURATION = std::chrono::microseconds(50);
115};
