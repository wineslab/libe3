49ErrorCode E3Agent::init() {
50 if (impl_->interface) {
51 E3_LOG_WARN(LOG_TAG) << "Agent already initialized";
52 return ErrorCode::ALREADY_INITIALIZED;
53 }
54
55 E3_LOG_INFO(LOG_TAG) << "Initializing E3Agent";
56
57 impl_->interface = std::make_unique<E3Interface>(impl_->config);
58
59 ErrorCode result = impl_->interface->init();
60 if (result != ErrorCode::SUCCESS) {
61 impl_->interface.reset();
62 E3_LOG_ERROR(LOG_TAG) << "Failed to initialize interface: "
63 << error_code_to_string(result);
64 return result;
65 }
66
67 if (impl_->dapp_report_handler) {
68 impl_->interface->set_dapp_report_handler(impl_->dapp_report_handler);
69 }
70
71 if (impl_->dapp_status_changed_handler) {
72 impl_->interface->set_dapp_status_changed_handler(impl_->dapp_status_changed_handler);
73 }
74
75 E3_LOG_INFO(LOG_TAG) << "E3Agent initialized successfully";
76 return ErrorCode::SUCCESS;
77}

79ErrorCode E3Agent::start() {
80 // Initialize if not already done
81 if (!impl_->interface) {
82 ErrorCode init_result = init();
83 if (init_result != ErrorCode::SUCCESS) {
84 return init_result;
85 }
86 }
87
88 E3_LOG_INFO(LOG_TAG) << "Starting E3Agent";
89
90 ErrorCode result = impl_->interface->start();
91 if (result != ErrorCode::SUCCESS) {
92 E3_LOG_ERROR(LOG_TAG) << "Failed to start interface: "
93 << error_code_to_string(result);
94 return result;
95 }
96
97 E3_LOG_INFO(LOG_TAG) << "E3Agent started successfully";
98 return ErrorCode::SUCCESS;
99}

167ErrorCode E3Agent::send_indication(
168 uint32_t dapp_id,
169 uint32_t ran_function_id,
170 const std::vector<uint8_t>& data
171) {
172 if (!impl_->interface || !impl_->interface->is_running()) {
173 return ErrorCode::NOT_INITIALIZED;
174 }
175
176 Pdu pdu(PduType::INDICATION_MESSAGE);
177 uint32_t mid = impl_->interface->generate_message_id();
178 pdu.message_id = mid;
179 IndicationMessage msg;
180 msg.dapp_identifier = dapp_id;
181 msg.ran_function_identifier = ran_function_id;
182 msg.protocol_data = data;
183 pdu.choice = msg;
184
185 return impl_->interface->queue_outbound(std::move(pdu));
186}

188ErrorCode E3Agent::send_xapp_control(
189 uint32_t dapp_id,
190 uint32_t ran_function_id,
191 const std::vector<uint8_t>& control_data
192) {
193 if (!impl_->interface || !impl_->interface->is_running()) {
194 return ErrorCode::NOT_INITIALIZED;
195 }
196 Pdu pdu(PduType::XAPP_CONTROL_ACTION);
197 pdu.message_id = impl_->interface->generate_message_id();
198 XAppControlAction action;
199 action.dapp_identifier = dapp_id;
200 action.ran_function_identifier = ran_function_id;
201 action.xapp_control_data = control_data;
202 pdu.choice = action;
203 return impl_->interface->queue_outbound(std::move(pdu));
204}

206ErrorCode E3Agent::send_message_ack(uint32_t request_id, ResponseCode response_code) {
207 if (!impl_->interface || !impl_->interface->is_running()) {
208 return ErrorCode::NOT_INITIALIZED;
209 }
210 Pdu pdu(PduType::MESSAGE_ACK);
211 pdu.message_id = impl_->interface->generate_message_id();
212 MessageAck ack;
213 ack.request_id = request_id;
214 ack.response_code = response_code;
215 pdu.choice = ack;
216 return impl_->interface->queue_outbound(std::move(pdu));
217}
