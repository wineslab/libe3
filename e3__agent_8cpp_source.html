57ErrorCode E3Agent::init() {
58 if (impl_->interface) {
59 E3_LOG_WARN(LOG_TAG) << "Agent already initialized";
60 return ErrorCode::ALREADY_INITIALIZED;
61 }
62
63 E3_LOG_INFO(LOG_TAG) << "Initializing E3Agent";
64
65 impl_->interface = std::make_unique<E3Interface>(impl_->config);
66
67 ErrorCode result = impl_->interface->init();
68 if (result != ErrorCode::SUCCESS) {
69 impl_->interface.reset();
70 E3_LOG_ERROR(LOG_TAG) << "Failed to initialize interface: "
71 << error_code_to_string(result);
72 return result;
73 }
74
75 if (impl_->dapp_report_handler) {
76 impl_->interface->set_dapp_report_handler(impl_->dapp_report_handler);
77 }
78
79 if (impl_->dapp_status_changed_handler) {
80 impl_->interface->set_dapp_status_changed_handler(impl_->dapp_status_changed_handler);
81 }
82
83 if (impl_->setup_response_handler) {
84 impl_->interface->set_setup_response_handler(impl_->setup_response_handler);
85 }
86 if (impl_->subscription_response_handler) {
87 impl_->interface->set_subscription_response_handler(impl_->subscription_response_handler);
88 }
89 if (impl_->indication_handler) {
90 impl_->interface->set_indication_handler(impl_->indication_handler);
91 }
92 if (impl_->xapp_control_handler) {
93 impl_->interface->set_xapp_control_handler(impl_->xapp_control_handler);
94 }
95 if (impl_->message_ack_handler) {
96 impl_->interface->set_message_ack_handler(impl_->message_ack_handler);
97 }
98
99 E3_LOG_INFO(LOG_TAG) << "E3Agent initialized successfully";
100 return ErrorCode::SUCCESS;
101}

103ErrorCode E3Agent::start() {
104 // Initialize if not already done
105 if (!impl_->interface) {
106 ErrorCode init_result = init();
107 if (init_result != ErrorCode::SUCCESS) {
108 return init_result;
109 }
110 }
111
112 E3_LOG_INFO(LOG_TAG) << "Starting E3Agent";
113
114 ErrorCode result = impl_->interface->start();
115 if (result != ErrorCode::SUCCESS) {
116 E3_LOG_ERROR(LOG_TAG) << "Failed to start interface: "
117 << error_code_to_string(result);
118 return result;
119 }
120
121 E3_LOG_INFO(LOG_TAG) << "E3Agent started successfully";
122 return ErrorCode::SUCCESS;
123}

195ErrorCode E3Agent::send_indication(
196 uint32_t dapp_id,
197 uint32_t ran_function_id,
198 const std::vector<uint8_t>& data
199) {
200 if (impl_->config.role != E3Role::RAN) {
201 return ErrorCode::STATE_ERROR;
202 }
203 if (!impl_->interface || !impl_->interface->is_running()) {
204 return ErrorCode::NOT_INITIALIZED;
205 }
206
207 Pdu pdu(PduType::INDICATION_MESSAGE);
208 uint32_t mid = impl_->interface->generate_message_id();
209 pdu.message_id = mid;
210 IndicationMessage msg;
211 msg.dapp_identifier = dapp_id;
212 msg.ran_function_identifier = ran_function_id;
213 msg.protocol_data = data;
214 pdu.choice = msg;
215
216 return impl_->interface->queue_outbound(std::move(pdu));
217}

220 uint32_t dapp_id,
221 uint32_t ran_function_id,
222 const std::vector<uint8_t>& control_data
223) {
224 if (impl_->config.role != E3Role::RAN) {
225 return ErrorCode::STATE_ERROR;
226 }
227 if (!impl_->interface || !impl_->interface->is_running()) {
228 return ErrorCode::NOT_INITIALIZED;
229 }
230 Pdu pdu(PduType::XAPP_CONTROL_ACTION);
231 pdu.message_id = impl_->interface->generate_message_id();
232 XAppControlAction action;
233 action.dapp_identifier = dapp_id;
234 action.ran_function_identifier = ran_function_id;
235 action.xapp_control_data = control_data;
236 pdu.choice = action;
237 return impl_->interface->queue_outbound(std::move(pdu));

286size_t E3Agent::dapp_count() const {
287 if (!impl_->interface) {
288 return 0;
289 }
