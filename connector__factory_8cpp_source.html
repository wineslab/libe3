25 E3Role role
26) {
27 std::unique_ptr<E3Connector> conn;

30 conn = std::make_unique<ZmqE3Connector>(
31 transport_layer, setup_endpoint, inbound_endpoint, outbound_endpoint,
32 setup_port, inbound_port, outbound_port, io_threads
33 );
34 break;
35
36 case E3LinkLayer::POSIX:
37 conn = std::make_unique<PosixE3Connector>(
38 transport_layer, setup_endpoint, inbound_endpoint, outbound_endpoint,
39 setup_port, inbound_port, outbound_port
40 );
41 break;
42
43 default:
44 E3_LOG_ERROR("ConnFactory") << "Unsupported link layer: "
45 << static_cast<int>(link_layer);
46 return nullptr;
47 }
48 if (conn) conn->set_role(role);
49 return conn;
