25// Human-readable name and the CMake flag that compiles in each format, used to
26// produce an actionable error when a requested encoding was not built in.
27struct EncodingInfo {
28 const char* name;
29 const char* enable_flag;
30};
31
32EncodingInfo encoding_info(EncodingFormat format) {
33 switch (format) {
34 case EncodingFormat::ASN1:
35 return {"ASN.1", "LIBE3_ENABLE_ASN1"};
36 case EncodingFormat::JSON:
37 return {"JSON", "LIBE3_ENABLE_JSON"};
38 case EncodingFormat::PROTOBUF:
39 return {"Protocol Buffers", "LIBE3_ENABLE_PROTOBUF"};
40 }
41 return {"unknown", nullptr};
42}
43} // namespace
44
45std::unique_ptr<E3Encoder> create_encoder(EncodingFormat format) {
46 switch (format) {
47#if LIBE3_ENABLE_ASN1
48 case EncodingFormat::ASN1:
49 return std::make_unique<Asn1E3Encoder>();
50#endif
51#if LIBE3_ENABLE_JSON
52 case EncodingFormat::JSON:
53 return std::make_unique<JsonE3Encoder>();
54#endif
55#if LIBE3_ENABLE_PROTOBUF
56 case EncodingFormat::PROTOBUF:
57 return std::make_unique<ProtobufE3Encoder>();
58#endif
59 default:
60 break;
61 }
62
63 // The requested format's case was either compiled out (its LIBE3_ENABLE_*
64 // flag was off in this build) or is not a known EncodingFormat value.
65 const EncodingInfo info = encoding_info(format);
66 if (info.enable_flag) {
67 E3_LOG_ERROR(LOG_TAG)
68 << "Encoding format " << info.name
69 << " requested but not compiled into this libe3 build; rebuild with -D"
70 << info.enable_flag << "=ON";
71 } else {
72 E3_LOG_ERROR(LOG_TAG) << "Unknown encoding format requested: "
73 << static_cast<unsigned>(format);
74 }
75 return nullptr;
76}
