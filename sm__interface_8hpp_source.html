28/**
29 * @brief Abstract Service Model interface
30 *
31 * This is the extension point for RAN vendors to implement their
32 * Service Models. The E3 agent treats E3SM as opaque - all SM-specific
33 * encoding/decoding happens in the implementations of this interface.
34 *
35 * RAN vendors should:
36 * 1. Derive from this class for each supported SM
37 * 2. Implement the required virtual methods
38 * 3. Register their SMs with the E3Agent
39 */
40class ServiceModel {
41public:
42 virtual ~ServiceModel() = default;
43
44 /**
45 * @brief Get SM name
46 */
47 virtual std::string name() const = 0;
48
49 /**
50 * @brief Get SM version
51 */
52 virtual uint32_t version() const = 0;
53
54 /**
55 * @brief Get RAN function ID for this SM
56 */
57 virtual uint32_t ran_function_id() const = 0;
58
59 /**
60 * @brief Get telemetry IDs supported by this SM
61 */
62 virtual std::vector<uint32_t> telemetry_ids() const = 0;
63
64 /**
65 * @brief Get control IDs supported by this SM
66 */
67 virtual std::vector<uint32_t> control_ids() const = 0;
68
69 /**
70 * @brief Optional RAN-function-specific opaque data
71 *
72 * Returns a byte vector that will be included in the SetupResponse
73 * as additional payload (ranFunctionList->ranFunctionData OCTET STRING for ASN).
74 * Default is empty since it is optional.
75 */
76 virtual std::vector<uint8_t> ran_function_data() const { return {}; }
77
78 /**
79 * @brief Initialize the SM
80 *
81 * Called when the SM is first registered.
82 * @return ErrorCode::SUCCESS on success
83 */
84 virtual ErrorCode init() = 0;
85
86 /**
87 * @brief Destroy the SM and release resources
88 *
89 * Called when the SM is being unregistered.
90 */
91 virtual void destroy() = 0;
92
93 /**
94 * @brief Start the SM processing
95 *
96 * Called when the first dApp subscribes to this SM's RAN function.
97 * @return ErrorCode::SUCCESS on success
98 */
99 virtual ErrorCode start() = 0;
100
101 /**
102 * @brief Stop the SM processing
103 *
104 * Called when the last dApp unsubscribes from this SM's RAN function.
105 */
106 virtual void stop() = 0;
107
108 /**
109 * @brief Check if SM is currently running
110 */
111 virtual bool is_running() const = 0;
112
113 /**
114 * @brief Process a control action from a dApp
115 *
116 * Called by the E3 interface when a DAppControlAction is received for this
117 * service model. Implementations can emit outbound PDUs (e.g. MessageAck,
118 * IndicationMessage) via emit_outbound().
119 */
120 virtual ErrorCode handle_control_action(
121 uint32_t request_message_id,
122 const DAppControlAction& action
123 ) = 0;
124
125protected:
126 ServiceModel() = default;
127
128 /**
129 * @brief Build a MessageAck PDU for a control/request message.
130 *
131 * This helper only builds the PDU. Implementations can decide whether to
132 * emit it with emit_outbound() or suppress it when protocol semantics allow.
133 */
134 static Pdu make_message_ack_pdu(
135 uint32_t request_message_id,
136 ResponseCode response_code
137 ) {
138 Pdu pdu(PduType::MESSAGE_ACK);
139 MessageAck ack;
140 ack.request_id = request_message_id;
141 ack.response_code = response_code;
142 pdu.choice = ack;
143 return pdu;
144 }
145
146 /**
147 * @brief Build an IndicationMessage PDU.
148 *
149 * This helper only builds the PDU. Implementations can emit it with
150 * emit_outbound() when appropriate.
151 */
152 static Pdu make_indication_pdu(
153 uint32_t dapp_id,
154 uint32_t ran_function_id,
155 std::vector<uint8_t> protocol_data
156 ) {
157 Pdu pdu(PduType::INDICATION_MESSAGE);
158 IndicationMessage msg;
159 msg.dapp_identifier = dapp_id;
160 msg.ran_function_identifier = ran_function_id;
161 msg.protocol_data = std::move(protocol_data);
162 pdu.choice = std::move(msg);
163 return pdu;
164 }
165
166 /**
167 * @brief Emit an outbound PDU produced by the SM.
168 */
169 ErrorCode emit_outbound(Pdu&& pdu) {
170 if (!outbound_emitter_) {
171 return ErrorCode::NOT_INITIALIZED;
172 }
173 return outbound_emitter_(std::move(pdu));
174 }
175
176 /**
177 * @brief Get currently subscribed dApps for this SM's RAN function.
178 */
179 std::vector<uint32_t> get_subscribers() const {
180 if (!subscribers_provider_) {
181 return {};
182 }
183 return subscribers_provider_();
184 }
185
186 /**
187 * @brief Set the outbound PDU emitter callback (used by E3Interface).
188 */
189 void set_outbound_emitter(std::function<ErrorCode(Pdu&&)> emitter) {
190 outbound_emitter_ = std::move(emitter);
191 }
192
193 /**
194 * @brief Set subscriber provider callback (used by E3Interface).
195 */
196 void set_subscribers_provider(std::function<std::vector<uint32_t>()> provider) {
197 subscribers_provider_ = std::move(provider);
198 }
199
200private:
201 std::function<ErrorCode(Pdu&&)> outbound_emitter_;
202 std::function<std::vector<uint32_t>()> subscribers_provider_;
203
204 friend class E3Interface;
205};
