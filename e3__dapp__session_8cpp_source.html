30struct DAppSession::Impl {
31 explicit Impl(libe3::E3Config config, std::size_t queue_capacity)
32 : agent(std::make_unique<libe3::E3Agent>(std::move(config))),
33 queue(queue_capacity) {}
34
35 std::unique_ptr<libe3::E3Agent> agent;
36 // Stable object (never moved) so a poll_events() reference stays valid; a
37 // prior stop() shut it down, and start() re-arms it in place (LockFreeQueue's
38 // shutdown is one-way, E3Agent itself is restartable).
39 libe3::LockFreeQueue<E3Event> queue;
40 std::atomic<unsigned long long> dropped{0};
41
42 // SetupResponse snapshot for the setup accessors (written once on setup).
43 mutable std::mutex setup_mu;
44 bool have_setup{false};
45 libe3::SetupResponse setup;
46
47 void enqueue(E3Event ev) {
48 if (queue.push(std::move(ev)) != libe3::ErrorCode::SUCCESS) {
49 dropped.fetch_add(1, std::memory_order_relaxed);
50 }
51 }
52};

54DAppSession::DAppSession(libe3::E3Config config, std::size_t queue_capacity) {
55 config.role = libe3::E3Role::DAPP; // this seam is dApp-only
56 impl_ = std::make_unique<Impl>(std::move(config), queue_capacity);
57 Impl* impl = impl_.get();
58 auto& agent = *impl->agent;
59
60 agent.set_indication_handler([impl](const libe3::IndicationMessage& m) {
61 E3Event ev;
62 ev.kind = E3_EVENT_INDICATION;
63 ev.dapp_id = m.dapp_identifier;
64 ev.ran_function_id = m.ran_function_identifier;
65 ev.payload = m.protocol_data; // one copy from libe3's const&
66 impl->enqueue(std::move(ev));
67 });
68
69 agent.set_xapp_control_handler([impl](const libe3::XAppControlAction& a) {
70 E3Event ev;
71 ev.kind = E3_EVENT_XAPP_CONTROL;
72 ev.dapp_id = a.dapp_identifier;
73 ev.ran_function_id = a.ran_function_identifier;
74 ev.payload = a.xapp_control_data;
75 impl->enqueue(std::move(ev));
76 });
77
78 agent.set_subscription_response_handler([impl](const libe3::SubscriptionResponse& r) {
79 E3Event ev;
80 ev.kind = E3_EVENT_SUBSCRIPTION_RESPONSE;
81 ev.dapp_id = r.dapp_identifier;
82 ev.request_id = r.request_id;
83 ev.subscription_id = r.subscription_id.value_or(0);
84 ev.response_code = static_cast<int>(r.response_code);
85 impl->enqueue(std::move(ev));
86 });
87
88 agent.set_setup_response_handler([impl](const libe3::SetupResponse& r) {
89 {
90 std::lock_guard<std::mutex> lk(impl->setup_mu);
91 impl->setup = r;
92 impl->have_setup = true;
93 }
94 E3Event ev;
95 ev.kind = E3_EVENT_SETUP_RESPONSE;
96 ev.dapp_id = r.dapp_identifier.value_or(0);
97 ev.request_id = r.request_id;
98 ev.response_code = static_cast<int>(r.response_code);
99 impl->enqueue(std::move(ev));
100 });
101
102 agent.set_message_ack_handler([impl](const libe3::MessageAck& ack) {
103 E3Event ev;
104 ev.kind = E3_EVENT_MESSAGE_ACK;
105 ev.request_id = ack.request_id;
106 ev.response_code = static_cast<int>(ack.response_code);
107 impl->enqueue(std::move(ev));
108 });
109}

118int DAppSession::start() {
119 // Re-arm the inbound ring in place (a prior stop() shut it down; E3Agent is
120 // restartable). Re-arming the same object — rather than replacing it — keeps
121 // any reference held by a concurrent poll_events() valid.
122 impl_->queue.rearm();
123 return static_cast<int>(impl_->agent->start());
124}

215std::vector<E3Event> DAppSession::poll_events(std::size_t max_batch, int timeout_ms) {
216 std::vector<E3Event> out;
217 if (max_batch == 0) return out;
218 auto& queue = impl_->queue;
219
220 // timeout_ms <= 0 is a non-blocking poll (try_pop); a positive value blocks
221 // up to that long for the first event. (pop() spins/yields before checking
222 // the deadline, so 0 there would busy-wait.)
223 std::optional<E3Event> first =
224 (timeout_ms <= 0) ? queue.try_pop()
225 : queue.pop(std::chrono::milliseconds(timeout_ms));
226 if (!first) return out; // quiet tick, or shutdown after stop()
227
228 out.reserve(std::min(max_batch, queue.capacity()));
229 out.push_back(std::move(*first));
230 while (out.size() < max_batch) {
231 auto next = queue.try_pop();
232 if (!next) break;
233 out.push_back(std::move(*next));
234 }
235 return out;
236}
