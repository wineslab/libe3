42ErrorCode E3Interface::init() {
43 if (state_.load() != AgentState::UNINITIALIZED) {
44 E3_LOG_WARN(LOG_TAG) << "Interface already initialized";
45 return ErrorCode::ALREADY_INITIALIZED;
46 }
47
48 E3_LOG_INFO(LOG_TAG) << "Initializing E3Interface";
49
50 // Create encoder
51 encoder_ = create_encoder(config_.encoding);
52 if (!encoder_) {
53 E3_LOG_ERROR(LOG_TAG) << "Failed to create encoder";
54 return ErrorCode::INTERNAL_ERROR;
55 }
56
57 // Create subscription manager
58 subscription_manager_ = std::make_unique<SubscriptionManager>();
59
60 // Set up SM lifecycle callback
61 subscription_manager_->set_sm_lifecycle_callback(
62 [this](uint32_t ran_function_id, bool should_start) {
63 on_sm_lifecycle_change(ran_function_id, should_start);
64 }
65 );
66
67 // Create response queue
68 response_queue_ = std::make_unique<ResponseQueue>();
69
70 // Create connector
71 connector_ = create_connector(
72 config_.link_layer,
73 config_.transport_layer,
74 config_.setup_endpoint,
75 config_.subscriber_endpoint,
76 config_.publisher_endpoint,
77 config_.setup_port,
78 config_.subscriber_port,
79 config_.publisher_port,
80 config_.io_threads
81 );
82
83 if (!connector_) {
84 E3_LOG_ERROR(LOG_TAG) << "Failed to create connector";
85 return ErrorCode::INTERNAL_ERROR;
86 }
87
88 state_.store(AgentState::INITIALIZED);
89 E3_LOG_INFO(LOG_TAG) << "E3Interface initialized successfully";
90
91 return ErrorCode::SUCCESS;
92}

94ErrorCode E3Interface::start() {
95 AgentState expected = AgentState::INITIALIZED;
96 if (!state_.compare_exchange_strong(expected, AgentState::CONNECTING)) {
97 E3_LOG_ERROR(LOG_TAG) << "Cannot start from state: "
98 << agent_state_to_string(expected);
99 return ErrorCode::STATE_ERROR;
100 }
101
102 E3_LOG_INFO(LOG_TAG) << "Starting E3Interface";
103 should_stop_.store(false);
104
105 // Set up initial connection
106 ErrorCode conn_result = connector_->setup_initial_connection();
107 if (conn_result != ErrorCode::SUCCESS) {
108 E3_LOG_ERROR(LOG_TAG) << "Failed to setup initial connection";
109 state_.store(AgentState::ERROR);
110 return conn_result;
111 }
112
113 state_.store(AgentState::CONNECTED);
114
115 // Start threads
116 setup_thread_ = std::make_unique<std::thread>(&E3Interface::setup_loop, this);
117 subscriber_thread_ = std::make_unique<std::thread>(&E3Interface::subscriber_loop, this);
118 publisher_thread_ = std::make_unique<std::thread>(&E3Interface::publisher_loop, this);
119
120 state_.store(AgentState::RUNNING);
121 E3_LOG_INFO(LOG_TAG) << "E3Interface started successfully";
122
123 return ErrorCode::SUCCESS;
124}

126void E3Interface::stop() {
127 if (state_.load() == AgentState::UNINITIALIZED ||
128 state_.load() == AgentState::STOPPING) {
129 return;
130 }
131
132 E3_LOG_INFO(LOG_TAG) << "Stopping E3Interface";
133 state_.store(AgentState::STOPPING);
134 should_stop_.store(true);
135
136 // Wake up response queue
137 if (response_queue_) {
138 response_queue_->shutdown();
139 }
140
141 // Interrupt blocking socket operations
142 if (connector_) {
143 connector_->shutdown();
144 }
145
146 // Join threads
147 if (setup_thread_ && setup_thread_->joinable()) {
148 setup_thread_->join();
149 }
150 if (subscriber_thread_ && subscriber_thread_->joinable()) {
151 subscriber_thread_->join();
152 }
153 if (publisher_thread_ && publisher_thread_->joinable()) {
154 publisher_thread_->join();
155 }
156 if (sm_data_thread_ && sm_data_thread_->joinable()) {
157 sm_data_thread_->join();
158 }
159
160 // Clean up SM registry
161 SmRegistry::instance().clear();
162
163 // Dispose connector
164 if (connector_) {
165 connector_->dispose();
166 }
167
168 state_.store(AgentState::INITIALIZED);
169 E3_LOG_INFO(LOG_TAG) << "E3Interface stopped";
170}

183ErrorCode E3Interface::register_sm(std::unique_ptr<ServiceModel> sm) {
184 if (!sm) {
185 return ErrorCode::INVALID_PARAM;
186 }
187 const uint32_t ran_function_id = sm->ran_function_id();
188 sm->set_subscribers_provider([this, ran_function_id]() {
189 if (!subscription_manager_) {
190 return std::vector<uint32_t>{};
191 }
192 return subscription_manager_->get_subscribed_dapps(ran_function_id);
193 });
194 sm->set_outbound_emitter([this](Pdu&& pdu) {
195 if (pdu.message_id == 0) {
196 pdu.message_id = generate_message_id();
197 }
198 return queue_outbound(std::move(pdu));
199 });
200 return SmRegistry::instance().register_sm(std::move(sm));

213void E3Interface::setup_loop() {
214 E3_LOG_INFO(LOG_TAG) << "Setup loop started";
215
216 auto available_ran_functions = SmRegistry::instance().get_available_ran_functions();
217
218 while (!should_stop_.load()) {
219 std::vector<uint8_t> buffer;
220 int ret = connector_->recv_setup_request(buffer);
221
222 if (ret <= 0) {
223 if (should_stop_.load()) break;
224 continue;
225 }
226
227 E3_LOG_INFO(LOG_TAG) << "Setup request received: " << ret << " bytes";
228
229 // Decode the setup request
230 auto decode_result = encoder_->decode(buffer.data(), static_cast<size_t>(ret));
231 if (!decode_result) {
232 E3_LOG_ERROR(LOG_TAG) << "Failed to decode setup request; ret=" << ret;
233 continue;
234 }
235
236 Pdu& pdu = *decode_result;
237 if (pdu.type != PduType::SETUP_REQUEST) {
238 E3_LOG_ERROR(LOG_TAG) << "Unexpected PDU type in setup: "
239 << pdu_type_to_string(pdu.type);
240 continue;
241 }
242
243 auto* request = std::get_if<SetupRequest>(&pdu.choice);
244 if (!request) {
245 E3_LOG_ERROR(LOG_TAG) << "Failed to get SetupRequest from PDU";
246 continue;
247 }
248
249 handle_setup_request(*request, pdu.message_id);
250 }
251
252 E3_LOG_INFO(LOG_TAG) << "Setup loop stopped";
253}

255void E3Interface::subscriber_loop() {
256 E3_LOG_INFO(LOG_TAG) << "Subscriber loop started";
257
258 ErrorCode result = connector_->setup_inbound_connection();
259 if (result != ErrorCode::SUCCESS) {
260 E3_LOG_ERROR(LOG_TAG) << "Failed to setup inbound connection";
261 return;
262 }
263
264 std::vector<uint8_t> buffer;
265
266 while (!should_stop_.load()) {
267 int ret = connector_->receive(buffer);
268
269 if (ret <= 0) {
270 if (should_stop_.load()) break;
271 continue;
272 }
273
274 auto decode_result = encoder_->decode(buffer.data(), static_cast<size_t>(ret));
275 if (!decode_result) {
276 E3_LOG_ERROR(LOG_TAG) << "Failed to decode PDU in subscriber";
277 continue;
278 }
279
280 Pdu& pdu = *decode_result;
281
282 switch (pdu.type) {
283 case PduType::SUBSCRIPTION_REQUEST: {
284 auto* request = std::get_if<SubscriptionRequest>(&pdu.choice);
285 if (request) {
286 handle_subscription_request(*request, pdu.message_id);
287 }
288 break;
289 }
290
291 case PduType::SUBSCRIPTION_DELETE: {
292 auto* del = std::get_if<SubscriptionDelete>(&pdu.choice);
293 if (del) {
294 handle_subscription_delete(*del, pdu.message_id);
295 }
296 break;
297 }
298
299 case PduType::DAPP_CONTROL_ACTION: {
300 auto* action = std::get_if<DAppControlAction>(&pdu.choice);
301 if (action) {
302 handle_control_action(*action, pdu.message_id);
303 }
304 break;
305 }
306
307 case PduType::DAPP_REPORT: {
308 auto* report = std::get_if<DAppReport>(&pdu.choice);
309 if (report) {
310 handle_dapp_report(*report);
311 }
312 break;
313 }
314
315 case PduType::RELEASE_MESSAGE: {
316 auto* release = std::get_if<ReleaseMessage>(&pdu.choice);
317 if (release) {
318 handle_release_message(*release);
319 }
320 break;
321 }
322
323 default:
324 E3_LOG_WARN(LOG_TAG) << "Received unexpected PDU type: "
325 << pdu_type_to_string(pdu.type);
326 break;
327 }
328 }
329
330 E3_LOG_INFO(LOG_TAG) << "Subscriber loop stopped";
331}

333void E3Interface::publisher_loop() {
334 E3_LOG_INFO(LOG_TAG) << "Publisher loop started";
335
336 // Ignore SIGPIPE to handle closed connections gracefully
337 signal(SIGPIPE, SIG_IGN);
338
339 // Retry outbound connection setup
340 ErrorCode result;
341 do {
342 std::this_thread::sleep_for(std::chrono::seconds(3));
343 if (should_stop_.load()) return;
344
345 E3_LOG_INFO(LOG_TAG) << "Trying to setup outbound connection";
346 result = connector_->setup_outbound_connection();
347 } while (result != ErrorCode::SUCCESS && !should_stop_.load());
348
349 if (result != ErrorCode::SUCCESS) {
350 E3_LOG_ERROR(LOG_TAG) << "Failed to setup outbound connection";
351 return;
352 }
353 E3_LOG_INFO(LOG_TAG) << "Outbound connection established";
354
355 while (!should_stop_.load()) {
356 // Pop from queue (blocking)
357 auto pdu_opt = response_queue_->pop(std::chrono::milliseconds(10));
358
359 if (!pdu_opt) {
360 continue;
361 }
362
363 // Encode and send
364 auto encode_result = encoder_->encode(*pdu_opt);
365 if (!encode_result) {
366 E3_LOG_ERROR(LOG_TAG) << "Failed to encode PDU for sending";
367 continue;
368 }
369
370 ErrorCode send_result = connector_->send(encode_result->buffer);
371 if (send_result != ErrorCode::SUCCESS) {
372 E3_LOG_ERROR(LOG_TAG) << "Failed to send PDU";
373 } else {
374 E3_LOG_DEBUG(LOG_TAG) << "Sent PDU: " << pdu_type_to_string(pdu_opt->type);
375 }
376 }
377
378 E3_LOG_INFO(LOG_TAG) << "Publisher loop stopped";
379}

381void E3Interface::sm_data_handler_loop() {
382 E3_LOG_INFO(LOG_TAG) << "SM data handler started";
383
384 while (!should_stop_.load()) {
385 bool data_processed = false;
386
387 // Get active RAN functions
388 auto ran_functions = subscription_manager_->get_active_ran_functions();
389
390 if (ran_functions.empty()) {
391 std::this_thread::sleep_for(std::chrono::seconds(1));
392 continue;
393 }
394
395 for (uint32_t ran_func : ran_functions) {
396 // Get subscribers for this RAN function
397 auto subscribers = subscription_manager_->get_subscribed_dapps(ran_func);
398
399 if (subscribers.empty()) {
400 continue;
401 }
402
403 // Get SM for this RAN function
404 ServiceModel* sm = SmRegistry::instance().get_by_ran_function(ran_func);
405
406 if (!sm || !sm->is_running()) {
407 continue;
408 }
409
410 // In a full implementation, we would poll the SM for indication data here
411 // For now, indication data is delivered through callbacks
412 }
413
414 if (!data_processed) {
415 std::this_thread::sleep_for(SM_POLL_INTERVAL);
416 }
417 }
418
419 E3_LOG_INFO(LOG_TAG) << "SM data handler stopped";
420}

426void E3Interface::handle_setup_request(const SetupRequest& request, uint32_t request_message_id) {
427 E3_LOG_INFO(LOG_TAG) << "Handling setup request from dApp '" << request.dapp_name
428 << "' (version=" << request.dapp_version
429 << ", vendor=" << request.vendor
430 << ", e3ap_version=" << request.e3ap_protocol_version << ")";
431
432 ResponseCode response_code = ResponseCode::NEGATIVE;
433 uint32_t assigned_dapp_id = 0;
434
435 // Register the dApp
436 auto [result, dapp_id] = subscription_manager_->register_dapp();
437 assigned_dapp_id = dapp_id;
438
439 if (result == ErrorCode::SUCCESS) {
440 response_code = ResponseCode::POSITIVE;
441 E3_LOG_INFO(LOG_TAG) << "dApp '" << request.dapp_name << "' registered with assigned ID " << assigned_dapp_id;
442 notify_dapp_status_changed();
443 } else {
444 E3_LOG_ERROR(LOG_TAG) << "Failed to register dApp '" << request.dapp_name << "': " << error_code_to_string(result);
445 }
446
447 // Create and send response
448 // Get available RAN functions and convert to RanFunctionDef list
449 auto available_ran_function_ids = SmRegistry::instance().get_available_ran_functions();
450 E3_LOG_DEBUG(LOG_TAG) << "Available RAN function ids count: " << available_ran_function_ids.size();
451 std::vector<RanFunctionDef> ran_function_list;
452 for (auto id : available_ran_function_ids) {
453 RanFunctionDef func;
454 func.ran_function_identifier = id;
455 ServiceModel* sm = SmRegistry::instance().get_by_ran_function(id);
456 if (sm) {
457 func.telemetry_identifier_list = sm->telemetry_ids();
458 func.control_identifier_list = sm->control_ids();
459 // Include optional RAN-function-specific opaque data provided by the SM
460 func.ran_function_data = sm->ran_function_data();
461 }
462 ran_function_list.push_back(func);
463 }
464 E3_LOG_DEBUG(LOG_TAG) << "Constructed ran_function_list size: " << ran_function_list.size();
465 for (const auto &rf : ran_function_list) {
466 E3_LOG_DEBUG(LOG_TAG) << " RAN function id=" << rf.ran_function_identifier
467 << ", telemetry_count=" << rf.telemetry_identifier_list.size()
468 << ", control_count=" << rf.control_identifier_list.size()
469 << ", ran_function_data_len=" << rf.ran_function_data.size();
470 }
471
472 auto encode_result = encoder_->encode_setup_response(
473 generate_message_id(),
474 request_message_id,
475 response_code,
476 config_.e3ap_version,
477 assigned_dapp_id, // dapp_identifier
478 config_.ran_identifier, // ran_identifier
479 ran_function_list
480 );
481
482 if (!encode_result) {
483 E3_LOG_ERROR(LOG_TAG) << "Failed to encode setup response for request id " << request_message_id;
484 return;
485 }
486
487 ErrorCode send_result = connector_->send_response(encode_result->buffer);
488 if (send_result != ErrorCode::SUCCESS) {
489 E3_LOG_ERROR(LOG_TAG) << "Failed to send setup response for request id " << request_message_id
490 << "; error=" << error_code_to_string(send_result);
491 } else {
492 E3_LOG_INFO(LOG_TAG) << "Sent setup response for request id " << request_message_id;
493 }
494}

496void E3Interface::handle_subscription_request(const SubscriptionRequest& request, uint32_t request_message_id) {
497 E3_LOG_INFO(LOG_TAG) << "Handling subscription request from dApp " << request.dapp_identifier
498 << " for RAN function " << request.ran_function_identifier;
499
500 ResponseCode response_code = ResponseCode::NEGATIVE;
501 uint32_t subscription_id = 1;
502
503 // Check if dApp is registered
504 if (!subscription_manager_->is_dapp_registered(request.dapp_identifier)) {
505 E3_LOG_ERROR(LOG_TAG) << "dApp " << request.dapp_identifier << " not registered";
506 } else {
507 auto [result, sub_id] = subscription_manager_->add_subscription(
508 request.dapp_identifier,
509 request.ran_function_identifier
510 );
511 subscription_id = sub_id;
512
513 if (result == ErrorCode::SUCCESS || result == ErrorCode::SUBSCRIPTION_EXISTS) {
514 response_code = ResponseCode::POSITIVE;
515 E3_LOG_INFO(LOG_TAG) << "Subscription added: dApp " << request.dapp_identifier
516 << " -> RAN function " << request.ran_function_identifier
517 << " (subscription_id=" << subscription_id << ")";
518 if (result == ErrorCode::SUCCESS) {
519 notify_dapp_status_changed();
520 }
521 } else {
522 E3_LOG_ERROR(LOG_TAG) << "Failed to add subscription: "
523 << error_code_to_string(result);
524 }
525 }
526
527 // Create and queue response
528 Pdu response_pdu(PduType::SUBSCRIPTION_RESPONSE);
529 SubscriptionResponse resp;
530 response_pdu.message_id = generate_message_id();
531 resp.request_id = request_message_id;
532 resp.dapp_identifier = request.dapp_identifier;
533 resp.response_code = response_code;
534 if (response_code == ResponseCode::POSITIVE) {
535 resp.subscription_id = subscription_id;
536 }
537 response_pdu.choice = resp;
538
539 queue_outbound(std::move(response_pdu));
540}

542void E3Interface::handle_subscription_delete(const SubscriptionDelete& del, uint32_t request_message_id) {
543 E3_LOG_INFO(LOG_TAG) << "Handling subscription delete from dApp " << del.dapp_identifier
544 << " for subscription " << del.subscription_id;
545
546 ResponseCode response_code = ResponseCode::NEGATIVE;
547
548 ErrorCode result = subscription_manager_->remove_subscription_by_id(
549 del.dapp_identifier,
550 del.subscription_id
551 );
552
553 if (result == ErrorCode::SUCCESS) {
554 response_code = ResponseCode::POSITIVE;
555 E3_LOG_INFO(LOG_TAG) << "Subscription " << del.subscription_id << " removed for dApp " << del.dapp_identifier;
556 notify_dapp_status_changed();
557 } else {
558 E3_LOG_ERROR(LOG_TAG) << "Failed to remove subscription: "
559 << error_code_to_string(result);
560 }
561
562 // Create and queue ack response
563 Pdu response_pdu(PduType::MESSAGE_ACK);
564 MessageAck ack;
565 response_pdu.message_id = generate_message_id();
566 ack.request_id = request_message_id;
567 ack.response_code = response_code;
568 response_pdu.choice = ack;
569
570 queue_outbound(std::move(response_pdu));
571}
