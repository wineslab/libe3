30/**
31 * @brief Apply CPU-affinity and niceness to the calling thread.
32 *
33 * Invoked at the very start of each I/O thread so that the settings take
34 * effect before any real work begins.
35 *
36 * @param affinity Logical CPU core to pin to, or -1 to skip pinning.
37 * @param niceness Nice value in [-20, 19], or 0 to skip.
38 */
39void apply_thread_config(int affinity, int niceness) noexcept {
40#ifdef __linux__
41 if (affinity >= 0) {
42 cpu_set_t cpuset;
43 CPU_ZERO(&cpuset);
44 CPU_SET(static_cast<size_t>(affinity), &cpuset);
45 // Non-fatal if this fails (e.g. core index out of range).
46 (void)pthread_setaffinity_np(pthread_self(),
47 sizeof(cpu_set_t), &cpuset);
48 }
49 if (niceness != 0) {
50 // Use the kernel TID so only this thread's scheduling priority changes.
51 // syscall(SYS_gettid) returns the kernel thread ID as a long; cast it
52 // to pid_t first, then to the id_t expected by setpriority(PRIO_PROCESS).
53 auto tid = static_cast<id_t>(static_cast<pid_t>(syscall(SYS_gettid)));
54 // Non-fatal: negative values require CAP_SYS_NICE.
55 (void)setpriority(PRIO_PROCESS, tid, niceness);
56 }
57#else
58 (void)affinity;
59 (void)niceness;
60#endif
61}

64uint32_t E3Interface::generate_message_id() {
65 thread_local std::mt19937 rng(std::random_device{}());
66 std::uniform_int_distribution<uint32_t> dist(1, 1000);
67 return dist(rng);
68}

70E3Interface::E3Interface(const E3Config& config)
71 : config_(config)
72{
73 // Suppress shared log-file writes when logging is disabled. This lets
74 // multiple E3Interface instances (e.g. a RAN-role and a dApp-role one in
75 // the same process — integration tests, the latency benchmark, or
76 // production multi-peer dApps) coexist without racing on the same file.
77 if (config.log_level > 0) {
78 const char* log_path = (config.role == E3Role::DAPP)
79 ? "/tmp/e3_dapp.log"
80 : "/tmp/e3_agent.log";
81 Logger::instance().set_log_file(log_path);
82 }
83 Logger::instance().set_level(config.log_level);
84 E3_LOG_INFO(LOG_TAG) << "E3Interface created (role=" << role_to_string(config.role) << ")";
85}

92ErrorCode E3Interface::init() {
93 if (state_.load() != AgentState::UNINITIALIZED) {
94 E3_LOG_WARN(LOG_TAG) << "Interface already initialized";
95 return ErrorCode::ALREADY_INITIALIZED;
96 }
97
98 E3_LOG_INFO(LOG_TAG) << "Initializing E3Interface";
99
100 // Create encoder
101 encoder_ = create_encoder(config_.encoding);
102 if (!encoder_) {
103 E3_LOG_ERROR(LOG_TAG) << "Failed to create encoder";
104 return ErrorCode::INTERNAL_ERROR;
105 }
106
107 // Allocate role-specific state. Exactly one is non-null.
108 if (config_.role == E3Role::RAN) {
109 subscription_manager_ = std::make_unique<SubscriptionManager>();
110 subscription_manager_->set_sm_lifecycle_callback(
111 [this](uint32_t ran_function_id, bool should_start) {
112 on_sm_lifecycle_change(ran_function_id, should_start);
113 }
114 );
115 } else {
116 dapp_state_ = std::make_unique<DAppSubscriptionState>();
117 }
118
119 // Create response queue
120 response_queue_ = std::make_unique<ResponseQueue>();
121
122 // Create connector
123 connector_ = create_connector(
124 config_.link_layer,
125 config_.transport_layer,
126 config_.setup_endpoint,
127 config_.subscriber_endpoint,
128 config_.publisher_endpoint,
129 config_.setup_port,
130 config_.subscriber_port,
131 config_.publisher_port,
132 config_.io_threads,
133 config_.role
134 );
135
136 if (!connector_) {
137 E3_LOG_ERROR(LOG_TAG) << "Failed to create connector";
138 return ErrorCode::INTERNAL_ERROR;
139 }
140
141 state_.store(AgentState::INITIALIZED);
142 E3_LOG_INFO(LOG_TAG) << "E3Interface initialized successfully";
143
144 return ErrorCode::SUCCESS;
145}

147ErrorCode E3Interface::start() {
148 AgentState expected = AgentState::INITIALIZED;
149 if (!state_.compare_exchange_strong(expected, AgentState::CONNECTING)) {
150 E3_LOG_ERROR(LOG_TAG) << "Cannot start from state: "
151 << agent_state_to_string(expected);
152 return ErrorCode::STATE_ERROR;
153 }
154
155 E3_LOG_INFO(LOG_TAG) << "Starting E3Interface (role=" << role_to_string(config_.role) << ")";
156 should_stop_.store(false);
157
158 // Set up the setup channel. RAN binds; dApp connects.
159 ErrorCode conn_result = (config_.role == E3Role::RAN)
160 ? connector_->setup_initial_connection()
161 : connector_->setup_initial_connection_client();
162 if (conn_result != ErrorCode::SUCCESS) {
163 E3_LOG_ERROR(LOG_TAG) << "Failed to setup initial connection: "
164 << error_code_to_string(conn_result);
165 state_.store(AgentState::ERROR);
166 return conn_result;
167 }
168
169 state_.store(AgentState::CONNECTED);
170
171 // Start threads. RAN runs setup+inbound+outbound+sm_data; dApp runs
172 // setup+inbound+outbound (no sm_data — dApps don't host SMs).
173 if (config_.role == E3Role::RAN) {
174 setup_thread_ = std::make_unique<std::thread>(&E3Interface::setup_loop_ran, this);
175 inbound_thread_ = std::make_unique<std::thread>(&E3Interface::inbound_loop_ran, this);
176 outbound_thread_ = std::make_unique<std::thread>(&E3Interface::outbound_loop_ran, this);
177 } else {
178 setup_thread_ = std::make_unique<std::thread>(&E3Interface::setup_loop_dapp, this);
179 inbound_thread_ = std::make_unique<std::thread>(&E3Interface::inbound_loop_dapp, this);
180 outbound_thread_ = std::make_unique<std::thread>(&E3Interface::outbound_loop_dapp, this);
181 }
182
183 state_.store(AgentState::RUNNING);
184 E3_LOG_INFO(LOG_TAG) << "E3Interface started successfully";
185
186 return ErrorCode::SUCCESS;
187}

189void E3Interface::stop() {
190 if (state_.load() == AgentState::UNINITIALIZED ||
191 state_.load() == AgentState::STOPPING) {
192 return;
193 }
194
195 E3_LOG_INFO(LOG_TAG) << "Stopping E3Interface";
196 state_.store(AgentState::STOPPING);
197 should_stop_.store(true);
198
199 // Wake up response queue
200 if (response_queue_) {
201 response_queue_->shutdown();
202 }
203
204 // Interrupt blocking socket operations
205 if (connector_) {
206 connector_->shutdown();
207 }
208
209 // Join threads
210 if (setup_thread_ && setup_thread_->joinable()) {
211 setup_thread_->join();
212 }
213 if (inbound_thread_ && inbound_thread_->joinable()) {
214 inbound_thread_->join();
215 }
216 if (outbound_thread_ && outbound_thread_->joinable()) {
217 outbound_thread_->join();
218 }
219 if (sm_data_thread_ && sm_data_thread_->joinable()) {
220 sm_data_thread_->join();
221 }
222
223 // Wake up anyone blocked in wait_for_setup so they don't hang.
224 {
225 std::lock_guard<std::mutex> lk(setup_complete_mu_);
226 setup_complete_ = true;
227 }
228 setup_complete_cv_.notify_all();
229
230 // Clean up SM registry — only the RAN role owns SMs. Doing this
231 // unconditionally would wipe a sibling RAN-role E3Interface's registered
232 // SMs in a two-roles-in-one-process scenario (integration tests, the
233 // latency benchmark, multi-peer dApps colocated with a RAN). The dApp
234 // role never registers an SM, so there's nothing to clear.
235 if (config_.role == E3Role::RAN) {
236 SmRegistry::instance().clear();
237 }
238
239 // Dispose connector
240 if (connector_) {
241 connector_->dispose();
242 }
243
244 state_.store(AgentState::INITIALIZED);
245 E3_LOG_INFO(LOG_TAG) << "E3Interface stopped";
246}

259ErrorCode E3Interface::register_sm(std::unique_ptr<ServiceModel> sm) {
260 if (!sm) {
261 return ErrorCode::INVALID_PARAM;
262 }
263 const uint32_t ran_function_id = sm->ran_function_id();
264 sm->set_subscribers_provider([this, ran_function_id]() {
265 if (!subscription_manager_) {
266 return std::vector<uint32_t>{};
267 }
268 return subscription_manager_->get_subscribed_dapps(ran_function_id);
269 });
270 sm->set_outbound_emitter([this](Pdu&& pdu) {
271 if (pdu.message_id == 0) {
272 pdu.message_id = generate_message_id();
273 }
274 return queue_outbound(std::move(pdu));
275 });
276 return SmRegistry::instance().register_sm(std::move(sm));
277}

289void E3Interface::setup_loop_ran() {
290 E3_LOG_INFO(LOG_TAG) << "Setup loop (RAN) started";
291
292 auto available_ran_functions = SmRegistry::instance().get_available_ran_functions();
293
294 while (!should_stop_.load()) {
295 std::vector<uint8_t> buffer;
296 int ret = connector_->recv_setup_request(buffer);
297
298 if (ret <= 0) {
299 if (should_stop_.load()) break;
300 continue;
301 }
302
303 E3_LOG_INFO(LOG_TAG) << "Setup request received: " << ret << " bytes";
304
305 // Decode the setup request
306 auto decode_result = encoder_->decode(buffer.data(), static_cast<size_t>(ret));
307 if (!decode_result) {
308 E3_LOG_ERROR(LOG_TAG) << "Failed to decode setup request; ret=" << ret;
309 continue;
310 }
311
312 Pdu& pdu = *decode_result;
313 if (pdu.type != PduType::SETUP_REQUEST) {
314 E3_LOG_ERROR(LOG_TAG) << "Unexpected PDU type in setup: "
315 << pdu_type_to_string(pdu.type);
316 continue;
317 }
318
319 auto* request = std::get_if<SetupRequest>(&pdu.choice);
320 if (!request) {
321 E3_LOG_ERROR(LOG_TAG) << "Failed to get SetupRequest from PDU";
322 continue;
323 }
324
325 handle_setup_request(*request, pdu.message_id);
326 }
327
328 E3_LOG_INFO(LOG_TAG) << "Setup loop (RAN) stopped";
329}

331void E3Interface::inbound_loop_ran() {
332 apply_thread_config(config_.io_thread_affinity, config_.io_thread_niceness);
333 E3_LOG_INFO(LOG_TAG) << "Inbound loop (RAN) started";
334
335 ErrorCode result = connector_->setup_inbound_connection();
336 if (result != ErrorCode::SUCCESS) {
337 E3_LOG_ERROR(LOG_TAG) << "Failed to setup inbound connection";
338 return;
339 }
340
341 std::vector<uint8_t> buffer;
342
343 while (!should_stop_.load()) {
344 int ret = connector_->receive(buffer);
345
346 if (ret <= 0) {
347 if (should_stop_.load()) break;
348 continue;
349 }
350
351 auto decode_result = encoder_->decode(buffer.data(), static_cast<size_t>(ret));
352 if (!decode_result) {
353 E3_LOG_ERROR(LOG_TAG) << "Failed to decode PDU in subscriber";
354 continue;
355 }
356
357 Pdu& pdu = *decode_result;
358
359 switch (pdu.type) {
360 case PduType::SUBSCRIPTION_REQUEST: {
361 auto* request = std::get_if<SubscriptionRequest>(&pdu.choice);
362 if (request) {
363 handle_subscription_request(*request, pdu.message_id);
364 }
365 break;
366 }
367
368 case PduType::SUBSCRIPTION_DELETE: {
369 auto* del = std::get_if<SubscriptionDelete>(&pdu.choice);
370 if (del) {
371 handle_subscription_delete(*del, pdu.message_id);
372 }
373 break;
374 }
375
376 case PduType::DAPP_CONTROL_ACTION: {
377 auto* action = std::get_if<DAppControlAction>(&pdu.choice);
378 if (action) {
379 handle_control_action(*action, pdu.message_id);
380 }
381 break;
382 }
383
384 case PduType::DAPP_REPORT: {
385 auto* report = std::get_if<DAppReport>(&pdu.choice);
386 if (report) {
387 handle_dapp_report(*report);
388 }
389 break;
390 }
391
392 case PduType::RELEASE_MESSAGE: {
393 auto* release = std::get_if<ReleaseMessage>(&pdu.choice);
394 if (release) {
395 handle_release_message(*release);
396 }
397 break;
398 }
399
400 default:
401 E3_LOG_WARN(LOG_TAG) << "Received unexpected PDU type: "
402 << pdu_type_to_string(pdu.type);
403 break;
404 }
405 }
406
407 E3_LOG_INFO(LOG_TAG) << "Inbound loop (RAN) stopped";
408}

410void E3Interface::outbound_loop_ran() {
411 apply_thread_config(config_.io_thread_affinity, config_.io_thread_niceness);
412 E3_LOG_INFO(LOG_TAG) << "Outbound loop (RAN) started";
413
414 // Ignore SIGPIPE to handle closed connections gracefully
415 signal(SIGPIPE, SIG_IGN);
416
417 // Set up the outbound (PUB) socket. Try immediately; retry with backoff
418 // only on failure. Putting an unconditional sleep before the first
419 // attempt introduces a multi-second startup latency that breaks
420 // integration tests and any colocated RAN/dApp pair.
421 ErrorCode result = connector_->setup_outbound_connection();
422 while (result != ErrorCode::SUCCESS && !should_stop_.load()) {
423 E3_LOG_WARN(LOG_TAG) << "Outbound connection setup failed ("
424 << error_code_to_string(result) << "), retrying in 1s";
425 std::this_thread::sleep_for(std::chrono::seconds(1));
426 if (should_stop_.load()) return;
427 result = connector_->setup_outbound_connection();
428 }
429
430 if (result != ErrorCode::SUCCESS) {
431 E3_LOG_ERROR(LOG_TAG) << "Failed to setup outbound connection";
432 return;
433 }
434 E3_LOG_INFO(LOG_TAG) << "Outbound connection established";
435
436 while (!should_stop_.load()) {
437 // Pop from queue (blocking)
438 auto pdu_opt = response_queue_->pop(std::chrono::milliseconds(10));
439
440 if (!pdu_opt) {
441 continue;
442 }
443
444 // Encode and send
445 auto encode_result = encoder_->encode(*pdu_opt);
446 if (!encode_result) {
447 E3_LOG_ERROR(LOG_TAG) << "Failed to encode PDU for sending";
448 continue;
449 }
450
451 ErrorCode send_result = connector_->send(encode_result->buffer);
452 if (send_result != ErrorCode::SUCCESS) {
453 E3_LOG_ERROR(LOG_TAG) << "Failed to send PDU";
454 } else {
455 E3_LOG_DEBUG(LOG_TAG) << "Sent PDU: " << pdu_type_to_string(pdu_opt->type);
456 }
457 }
458
459 E3_LOG_INFO(LOG_TAG) << "Outbound loop (RAN) stopped";
460}

462void E3Interface::sm_data_handler_loop() {
463 E3_LOG_INFO(LOG_TAG) << "SM data handler started";
464
465 while (!should_stop_.load()) {
466 bool data_processed = false;
467
468 // Get active RAN functions
469 auto ran_functions = subscription_manager_->get_active_ran_functions();
470
471 if (ran_functions.empty()) {
472 std::this_thread::sleep_for(std::chrono::seconds(1));
473 continue;
474 }
475
476 for (uint32_t ran_func : ran_functions) {
477 // Get subscribers for this RAN function
478 auto subscribers = subscription_manager_->get_subscribed_dapps(ran_func);
479
480 if (subscribers.empty()) {
481 continue;
482 }
483
484 // Get SM for this RAN function
485 ServiceModel* sm = SmRegistry::instance().get_by_ran_function(ran_func);
486
487 if (!sm || !sm->is_running()) {
488 continue;
489 }
490
491 // In a full implementation, we would poll the SM for indication data here
492 // For now, indication data is delivered through callbacks
493 }
494
495 if (!data_processed) {
496 std::this_thread::sleep_for(SM_POLL_INTERVAL);
497 }
498 }
499
500 E3_LOG_INFO(LOG_TAG) << "SM data handler stopped";
501}

507void E3Interface::handle_setup_request(const SetupRequest& request, uint32_t request_message_id) {
508 E3_LOG_INFO(LOG_TAG) << "Handling setup request from dApp '" << request.dapp_name
509 << "' (version=" << request.dapp_version
510 << ", vendor=" << request.vendor
511 << ", e3ap_version=" << request.e3ap_protocol_version << ")";
512
513 ResponseCode response_code = ResponseCode::NEGATIVE;
514 uint32_t assigned_dapp_id = 0;
515
516 // Register the dApp
517 auto [result, dapp_id] = subscription_manager_->register_dapp();
518 assigned_dapp_id = dapp_id;
519
520 if (result == ErrorCode::SUCCESS) {
521 response_code = ResponseCode::POSITIVE;
522 E3_LOG_INFO(LOG_TAG) << "dApp '" << request.dapp_name << "' registered with assigned ID " << assigned_dapp_id;
523 notify_dapp_status_changed();
524 } else {
525 E3_LOG_ERROR(LOG_TAG) << "Failed to register dApp '" << request.dapp_name << "': " << error_code_to_string(result);
526 }
527
528 // Create and send response
529 // Get available RAN functions and convert to RanFunctionDef list
530 auto available_ran_function_ids = SmRegistry::instance().get_available_ran_functions();
531 E3_LOG_DEBUG(LOG_TAG) << "Available RAN function ids count: " << available_ran_function_ids.size();
532 std::vector<RanFunctionDef> ran_function_list;
533 for (auto id : available_ran_function_ids) {
534 RanFunctionDef func;
535 func.ran_function_identifier = id;
536 ServiceModel* sm = SmRegistry::instance().get_by_ran_function(id);
537 if (sm) {
538 func.telemetry_identifier_list = sm->telemetry_ids();
539 func.control_identifier_list = sm->control_ids();
540 // Include optional RAN-function-specific opaque data provided by the SM
541 func.ran_function_data = sm->ran_function_data();
542 }
543 ran_function_list.push_back(func);
544 }
545 E3_LOG_DEBUG(LOG_TAG) << "Constructed ran_function_list size: " << ran_function_list.size();
546 for (const auto &rf : ran_function_list) {
547 E3_LOG_DEBUG(LOG_TAG) << " RAN function id=" << rf.ran_function_identifier
548 << ", telemetry_count=" << rf.telemetry_identifier_list.size()
549 << ", control_count=" << rf.control_identifier_list.size()
550 << ", ran_function_data_len=" << rf.ran_function_data.size();
551 }
552
553 auto encode_result = encoder_->encode_setup_response(
554 generate_message_id(),
555 request_message_id,
556 response_code,
557 config_.e3ap_version,
558 assigned_dapp_id, // dapp_identifier
559 config_.ran_identifier, // ran_identifier
560 ran_function_list
561 );
562
563 if (!encode_result) {
564 E3_LOG_ERROR(LOG_TAG) << "Failed to encode setup response for request id " << request_message_id;
565 return;
566 }
567
568 ErrorCode send_result = connector_->send_response(encode_result->buffer);
569 if (send_result != ErrorCode::SUCCESS) {
570 E3_LOG_ERROR(LOG_TAG) << "Failed to send setup response for request id " << request_message_id
571 << "; error=" << error_code_to_string(send_result);
572 } else {
573 E3_LOG_INFO(LOG_TAG) << "Sent setup response for request id " << request_message_id;
574 }
575}

577void E3Interface::handle_subscription_request(const SubscriptionRequest& request, uint32_t request_message_id) {
578 E3_LOG_INFO(LOG_TAG) << "Handling subscription request from dApp " << request.dapp_identifier
579 << " for RAN function " << request.ran_function_identifier;
580
581 ResponseCode response_code = ResponseCode::NEGATIVE;
582 uint32_t subscription_id = 1;
583
584 // Check if dApp is registered
585 if (!subscription_manager_->is_dapp_registered(request.dapp_identifier)) {
586 E3_LOG_ERROR(LOG_TAG) << "dApp " << request.dapp_identifier << " not registered";
587 } else {
588 auto [result, sub_id] = subscription_manager_->add_subscription(
589 request.dapp_identifier,
590 request.ran_function_identifier
591 );
592 subscription_id = sub_id;
593
594 if (result == ErrorCode::SUCCESS || result == ErrorCode::SUBSCRIPTION_EXISTS) {
595 response_code = ResponseCode::POSITIVE;
596 E3_LOG_INFO(LOG_TAG) << "Subscription added: dApp " << request.dapp_identifier
597 << " -> RAN function " << request.ran_function_identifier
598 << " (subscription_id=" << subscription_id << ")";
599 if (result == ErrorCode::SUCCESS) {
600 notify_dapp_status_changed();
601 }
602 } else {
603 E3_LOG_ERROR(LOG_TAG) << "Failed to add subscription: "
604 << error_code_to_string(result);
605 }
606 }
607
608 // Create and queue response
609 Pdu response_pdu(PduType::SUBSCRIPTION_RESPONSE);
610 SubscriptionResponse resp;
611 response_pdu.message_id = generate_message_id();
612 resp.request_id = request_message_id;
613 resp.dapp_identifier = request.dapp_identifier;
614 resp.response_code = response_code;
615 if (response_code == ResponseCode::POSITIVE) {
616 resp.subscription_id = subscription_id;
617 }
618 response_pdu.choice = resp;
619
620 queue_outbound(std::move(response_pdu));
621}

623void E3Interface::handle_subscription_delete(const SubscriptionDelete& del, uint32_t request_message_id) {
624 E3_LOG_INFO(LOG_TAG) << "Handling subscription delete from dApp " << del.dapp_identifier
625 << " for subscription " << del.subscription_id;
626
627 ResponseCode response_code = ResponseCode::NEGATIVE;
628
629 ErrorCode result = subscription_manager_->remove_subscription_by_id(
630 del.dapp_identifier,
631 del.subscription_id
632 );
633
634 if (result == ErrorCode::SUCCESS) {
635 response_code = ResponseCode::POSITIVE;
636 E3_LOG_INFO(LOG_TAG) << "Subscription " << del.subscription_id << " removed for dApp " << del.dapp_identifier;
637 notify_dapp_status_changed();
638 } else {
639 E3_LOG_ERROR(LOG_TAG) << "Failed to remove subscription: "
640 << error_code_to_string(result);
641 }
642
643 // Create and queue ack response
644 Pdu response_pdu(PduType::MESSAGE_ACK);
645 MessageAck ack;
646 response_pdu.message_id = generate_message_id();
647 ack.request_id = request_message_id;
648 ack.response_code = response_code;
649 response_pdu.choice = ack;
650
651 queue_outbound(std::move(response_pdu));
652}

720void E3Interface::setup_loop_dapp() {
721 E3_LOG_INFO(LOG_TAG) << "Setup loop (dApp) started";
722
723 // Encode the SetupRequest from config_'s dApp identification fields.
724 auto enc = encoder_->encode_setup_request(
725 generate_message_id(),
726 config_.e3ap_version,
727 config_.dapp_name,
728 config_.dapp_version,
729 config_.vendor
730 );
731 if (!enc) {
732 E3_LOG_ERROR(LOG_TAG) << "Failed to encode SetupRequest";
733 std::lock_guard<std::mutex> lk(setup_complete_mu_);
734 setup_complete_ = true;
735 setup_succeeded_ = false;
736 setup_complete_cv_.notify_all();
737 return;
738 }
739
740 // Lazy-pirate REQ: send, wait for reply with timeout, retry on no reply.
741 constexpr int MAX_RETRIES = 5;
742 int retries = 0;
743 bool got_response = false;
744 while (!should_stop_.load() && retries < MAX_RETRIES) {
745 ErrorCode rc = connector_->send_setup_request_client(enc->buffer);
746 if (rc != ErrorCode::SUCCESS) {
747 E3_LOG_ERROR(LOG_TAG) << "send_setup_request_client failed: "
748 << error_code_to_string(rc);
749 ++retries;
750 std::this_thread::sleep_for(std::chrono::seconds(1));
751 continue;
752 }
753
754 // Try a few times to receive (connector recv has its own short timeout).
755 for (int attempt = 0; attempt < 4 && !should_stop_.load(); ++attempt) {
756 std::vector<uint8_t> resp_buf;
757 int n = connector_->recv_setup_response_client(resp_buf);
758 if (n > 0) {
759 auto decoded = encoder_->decode(resp_buf.data(), static_cast<size_t>(n));
760 if (!decoded) {
761 E3_LOG_ERROR(LOG_TAG) << "Failed to decode SetupResponse";
762 break;
763 }
764 if (decoded->type != PduType::SETUP_RESPONSE) {
765 E3_LOG_ERROR(LOG_TAG) << "Unexpected PDU on setup channel: "
766 << pdu_type_to_string(decoded->type);
767 break;
768 }
769 auto* resp = std::get_if<SetupResponse>(&decoded->choice);
770 if (resp) {
771 handle_setup_response(*resp);
772 got_response = true;
773 }
774 break;
775 }
776 if (n < 0) {
777 E3_LOG_ERROR(LOG_TAG) << "recv_setup_response_client error";
778 break;
779 }
780 }
781 if (got_response) break;
782 ++retries;
783 }
784
785 {
786 std::lock_guard<std::mutex> lk(setup_complete_mu_);
787 setup_complete_ = true;
788 setup_succeeded_ = got_response;
789 }
790 setup_complete_cv_.notify_all();
791
792 E3_LOG_INFO(LOG_TAG) << "Setup loop (dApp) finished, success="
793 << (got_response ? "yes" : "no");
794}

796void E3Interface::inbound_loop_dapp() {
797 apply_thread_config(config_.io_thread_affinity, config_.io_thread_niceness);
798 E3_LOG_INFO(LOG_TAG) << "Inbound loop (dApp) started";
799
800 ErrorCode result = connector_->setup_inbound_connection_client();
801 if (result != ErrorCode::SUCCESS) {
802 E3_LOG_ERROR(LOG_TAG) << "Failed to setup inbound (dApp) connection: "
803 << error_code_to_string(result);
804 return;
805 }
806
807 std::vector<uint8_t> buffer;
808 while (!should_stop_.load()) {
809 int ret = connector_->receive(buffer);
810 if (ret <= 0) {
811 if (should_stop_.load()) break;
812 continue;
813 }
814 auto decoded = encoder_->decode(buffer.data(), static_cast<size_t>(ret));
815 if (!decoded) {
816 E3_LOG_ERROR(LOG_TAG) << "Failed to decode PDU in dApp inbound";
817 continue;
818 }
819 Pdu& pdu = *decoded;
820 switch (pdu.type) {
821 case PduType::SUBSCRIPTION_RESPONSE: {
822 auto* r = std::get_if<SubscriptionResponse>(&pdu.choice);
823 if (r) handle_subscription_response(*r);
824 break;
825 }
826 case PduType::INDICATION_MESSAGE: {
827 auto* m = std::get_if<IndicationMessage>(&pdu.choice);
828 if (m) handle_indication(*m);
829 break;
830 }
831 case PduType::XAPP_CONTROL_ACTION: {
832 auto* a = std::get_if<XAppControlAction>(&pdu.choice);
833 if (a) handle_xapp_control_action(*a);
834 break;
835 }
836 case PduType::MESSAGE_ACK: {
837 auto* a = std::get_if<MessageAck>(&pdu.choice);
838 if (a) handle_message_ack(*a);
839 break;
840 }
841 default:
842 E3_LOG_WARN(LOG_TAG) << "dApp received unexpected PDU type: "
843 << pdu_type_to_string(pdu.type);
844 break;
845 }
846 }
847
848 E3_LOG_INFO(LOG_TAG) << "Inbound loop (dApp) stopped";
849}

851void E3Interface::outbound_loop_dapp() {
852 apply_thread_config(config_.io_thread_affinity, config_.io_thread_niceness);
853 E3_LOG_INFO(LOG_TAG) << "Outbound loop (dApp) started";
854
855 signal(SIGPIPE, SIG_IGN);
856
857 ErrorCode result = connector_->setup_outbound_connection_client();
858 if (result != ErrorCode::SUCCESS) {
859 E3_LOG_ERROR(LOG_TAG) << "Failed to setup outbound (dApp) connection: "
860 << error_code_to_string(result);
861 return;
862 }
863
864 while (!should_stop_.load()) {
865 auto pdu_opt = response_queue_->pop(std::chrono::milliseconds(10));
866 if (!pdu_opt) continue;
867
868 auto enc = encoder_->encode(*pdu_opt);
869 if (!enc) {
870 E3_LOG_ERROR(LOG_TAG) << "Failed to encode dApp outbound PDU";
871 continue;
872 }
873 ErrorCode rc = connector_->send(enc->buffer);
874 if (rc != ErrorCode::SUCCESS) {
875 E3_LOG_ERROR(LOG_TAG) << "Failed to send dApp outbound PDU";
876 } else {
877 E3_LOG_DEBUG(LOG_TAG) << "Sent dApp outbound PDU: "
878 << pdu_type_to_string(pdu_opt->type);
879 }
880 }
881
882 E3_LOG_INFO(LOG_TAG) << "Outbound loop (dApp) stopped";
883}

901void E3Interface::handle_subscription_response(const SubscriptionResponse& resp) {
902 if (dapp_state_) {
903 std::lock_guard<std::mutex> lk(dapp_state_->mu);
904 if (!dapp_state_->assigned_dapp_id.has_value() ||
905 *dapp_state_->assigned_dapp_id != resp.dapp_identifier) {
906 return; // not for us
907 }
908 }
909 E3_LOG_INFO(LOG_TAG) << "Handling SubscriptionResponse for request "
910 << resp.request_id << " rc="
911 << response_code_to_string(resp.response_code);
912 if (dapp_state_ && resp.response_code == ResponseCode::POSITIVE
913 && resp.subscription_id.has_value()) {
914 // We need the RAN function id; find it via the request_id mapping.
915 // Since we don't separately track requested rf_id at this layer yet,
916 // the SubscriptionResponse currently doesn't echo it. The dApp-facing
917 // layer (E3Agent::subscribe) is responsible for resolving rf_id and
918 // calling record_subscription if needed via handlers.
919 // For now we just record the sub_id without an rf_id mapping if
920 // we can't determine it; the user's handler can supplement.
921 }
922 if (subscription_response_handler_) {
923 subscription_response_handler_(resp);
