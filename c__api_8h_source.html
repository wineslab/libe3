65/**
66 * @brief Descriptor describing a ServiceModel implemented in C.
67 *
68 * - `name`, `version` and `ran_function_id` identify the SM.
69 * - `telemetry_ids` and `control_ids` point to arrays that will be copied
70 * by the implementation; the caller may free them after creating the SM.
71 * - Callbacks are invoked by the C++ adapter and receive `user_data`.
72 */
73typedef struct {
74 const char* name; // Service Model name (null-terminated string)
75 uint32_t version; // Service Model version
76 uint32_t ran_function_id; // RAN function ID for this SM
77
78 const uint32_t* telemetry_ids; // Array of supported telemetry IDs (copied by library)
79 size_t telemetry_ids_len; // Number of telemetry IDs
80
81 const uint32_t* control_ids; // Array of supported control IDs (copied by library)
82 size_t control_ids_len; // Number of control IDs
83
84 /* Optional opaque RAN function data (E3SM-encoded, sent in setup response) */
85 const uint8_t* ran_function_data; // Pointer to opaque RAN function data (may be NULL)
86 size_t ran_function_data_len; // Length of ran_function_data in bytes (0 if none)
87
88 /* Callbacks */
89 e3_sm_init_cb sm_init; // Called on SM registration
90 e3_sm_destroy_cb sm_destroy; // Called on SM destruction
91 e3_sm_start_cb sm_start; // Called when SM should start
92 e3_sm_stop_cb sm_stop; // Called when SM should stop
93 e3_sm_is_running_cb sm_is_running; // Called to query running state
94 e3_sm_process_control_cb sm_process_control; // Called to process control actions
95
96 void* sm_context; // Opaque pointer passed to all callbacks
97} e3_c_service_model_desc_t;

146/**
147 * @brief Configuration for creating an E3Agent from C.
148 *
149 * All string pointers are copied by the library; the caller may free them
150 * after the call. Use NULL for optional strings to keep library defaults.
151 *
152 * Enum values: link_layer: 0=ZMQ, 1=POSIX. transport_layer: 0=SCTP, 1=TCP, 2=IPC.
153 * encoding: 0=ASN1, 1=JSON. Use -1 for link_layer, transport_layer, encoding, log_level
154 * to keep defaults. Use 0 for setup_port, subscriber_port, publisher_port to keep defaults.
155 */
156typedef struct {
157 const char* ran_identifier;
158 int link_layer; /* 0=ZMQ, 1=POSIX, -1=default */
159 int transport_layer; /* 0=SCTP, 1=TCP, 2=IPC, -1=default */
160 uint16_t setup_port;
161 uint16_t subscriber_port;
162 uint16_t publisher_port;
163 const char* setup_endpoint;
164 const char* subscriber_endpoint;
165 const char* publisher_endpoint;
166 int encoding; /* 0=ASN1, 1=JSON, -1=default */
167 size_t io_threads; /* 0=default */
168 int log_level; /* -1=default */
169 const char* log_path; /* NULL=auto (/tmp/e3_<role>[_<id>]_<euid>.log) */
170} e3_config_t;
