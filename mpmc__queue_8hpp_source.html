1/**
2 * @file mpmc_queue.hpp
3 * @brief Lock-free bounded Multi-Producer Multi-Consumer (MPMC) ring buffer
4 *
5 * Implements Dmitry Vyukov's MPMC bounded queue algorithm. Each slot carries
6 * an atomic sequence counter that acts as a generation tag, eliminating the
7 * need for CAS on the data itself and making the queue safe for any number of
8 * concurrent producers and consumers with very low overhead.
9 *
10 * Design highlights:
11 * - Zero locks / zero condition variables in the fast path
12 * - head_ and tail_ are on separate cache lines to eliminate false sharing
13 * - Capacity is rounded up to the nearest power of two for cheap masking
14 * - Provides try_push / try_pop (non-blocking) used by LockFreeQueue
15 *
16 * SPDX-License-Identifier: Apache-2.0
17 */
18

31/**
32 * @brief Lock-free bounded MPMC ring buffer.
33 *
34 * @tparam T Element type. Must be default-constructible and movable.
35 *
36 * Usage:
37 * @code
38 * MpmcQueue<MyMsg> q(256);
39 * // producer:
40 * if (!q.try_push(std::move(msg))) { // queue full }
41 * // consumer:
42 * MyMsg out;
43 * if (q.try_pop(out)) { // use out }
44 * @endcode
45 */
46template<typename T>
47class MpmcQueue {
48public:
49 /**
50 * @brief Construct a queue with at least @p min_capacity slots.
51 *
52 * The actual capacity is rounded up to the next power of two (minimum 2).
53 * @param min_capacity Minimum number of slots (0 is treated as 1).
54 */
55 explicit MpmcQueue(size_t min_capacity) {
56 // Round up to next power of two (minimum 2 to make the mask sensible)
57 capacity_ = 2;
58 while (capacity_ < min_capacity) capacity_ <<= 1;
59 mask_ = capacity_ - 1;
60
61 // Allocate cache-line aligned buffer to prevent false sharing between
62 // adjacent slots' sequence counters on small-element queues.
63 buffer_ = static_cast<Slot*>(
64 ::operator new(sizeof(Slot) * capacity_,
65 std::align_val_t{CACHE_LINE_SIZE})
66 );
67
68 for (size_t i = 0; i < capacity_; ++i) {
69 new (&buffer_[i]) Slot();
70 // Sequence == index means "slot is free for the next producer
71 // whose enqueue position is i".
72 buffer_[i].sequence.store(i, std::memory_order_relaxed);
73 }
74 }
75
76 ~MpmcQueue() {
77 for (size_t i = 0; i < capacity_; ++i) {
78 buffer_[i].~Slot();
79 }
80 ::operator delete(buffer_, std::align_val_t{CACHE_LINE_SIZE});
81 }

89 /**
90 * @brief Try to enqueue an item (non-blocking, move semantics).
91 * @return true on success; false if the queue is full.
92 */
93 bool try_push(T&& data) {
94 size_t pos = head_.load(std::memory_order_relaxed);
95 for (;;) {
96 Slot& slot = buffer_[pos & mask_];
97 size_t seq = slot.sequence.load(std::memory_order_acquire);
98 intptr_t diff = static_cast<intptr_t>(seq)
99 - static_cast<intptr_t>(pos);
100
101 if (diff == 0) {
102 // Slot is ready for this producer; claim it.
103 if (head_.compare_exchange_weak(pos, pos + 1,
104 std::memory_order_relaxed)) {
105 slot.data = std::move(data);
106 // Mark slot as containing valid data for consumer at pos+1.
107 slot.sequence.store(pos + 1, std::memory_order_release);
108 return true;
109 }
110 // Another producer claimed this slot; reload and retry.
111 } else if (diff < 0) {
112 return false; // Queue is full
113 } else {
114 // Another producer advanced head past us; reload.
115 pos = head_.load(std::memory_order_relaxed);
116 }
117 }
118 }
119
120 /**
121 * @brief Try to enqueue an item (copy overload).
122 */
123 bool try_push(const T& data) {
124 T copy = data;
125 return try_push(std::move(copy));
126 }

128 /**
129 * @brief Try to dequeue an item (non-blocking).
130 * @param[out] data Receives the dequeued item on success.
131 * @return true on success; false if the queue is empty.
132 */
133 bool try_pop(T& data) {
134 size_t pos = tail_.load(std::memory_order_relaxed);
135 for (;;) {
136 Slot& slot = buffer_[pos & mask_];
137 size_t seq = slot.sequence.load(std::memory_order_acquire);
138 intptr_t diff = static_cast<intptr_t>(seq)
139 - static_cast<intptr_t>(pos + 1);
140
141 if (diff == 0) {
142 // Slot holds data for this consumer; claim it.
143 if (tail_.compare_exchange_weak(pos, pos + 1,
144 std::memory_order_relaxed)) {
145 data = std::move(slot.data);
146 // Recycle slot for the producer that wraps around.
147 slot.sequence.store(pos + capacity_,
148 std::memory_order_release);
149 return true;
150 }
151 // Another consumer claimed this slot; reload and retry.
152 } else if (diff < 0) {
153 return false; // Queue is empty
154 } else {
155 // Another consumer advanced tail past us; reload.
156 pos = tail_.load(std::memory_order_relaxed);
157 }
158 }
159 }

161 /**
162 * @brief Approximate number of items currently in the queue.
163 *
164 * Due to concurrent modifications this value may be slightly stale,
165 * but is always in [0, capacity()].
166 */
167 size_t size_approx() const noexcept {
168 size_t h = head_.load(std::memory_order_relaxed);
169 size_t t = tail_.load(std::memory_order_relaxed);
170 return h >= t ? h - t : 0;
171 }
172
173 /** @brief Return true if the queue appears empty (approximate). */
174 bool empty_approx() const noexcept { return size_approx() == 0; }
175
176 /** @brief Actual slot capacity (always a power of two). */
177 size_t capacity() const noexcept { return capacity_; }
178
179private:
180 struct Slot {
181 std::atomic<size_t> sequence{0};
182 T data;
183 };
184
185 Slot* buffer_{nullptr};
186 size_t capacity_{0};
187 size_t mask_{0};
188
189 // Separate cache lines for head and tail to prevent producer/consumer
190 // false sharing.
191 alignas(CACHE_LINE_SIZE) std::atomic<size_t> head_{0};
192 alignas(CACHE_LINE_SIZE) std::atomic<size_t> tail_{0};
193};

195/**
196 * @brief Emit a CPU-level pause/yield hint inside spin-wait loops.
197 *
198 * Reduces power consumption and inter-core bus traffic while spinning.
199 * On architectures without a dedicated pause instruction this is a no-op.
200 */
201inline void cpu_relax() noexcept {
202#if defined(__x86_64__) || defined(__i386__)
203 __asm__ volatile("pause" ::: "memory");
204#elif defined(__aarch64__) || defined(__arm__)
205 __asm__ volatile("yield" ::: "memory");
206#endif
207}
