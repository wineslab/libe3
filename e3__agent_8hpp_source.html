32/**
33 * @brief E3Agent - Main façade for RAN vendor integration
34 *
35 * This class is the primary entry point for integrating E3AP functionality
36 * into a RAN function (DU, CU-CP, CU-UP). It provides:
37 *
38 * - Simple lifecycle management (init, start, stop, destroy)
39 * - Service Model registration
40 * - Automatic routing of control actions to the correct Service Model
41 * - Thread-safe operation
42 *
43 * **Usage Example:**
44 * ```cpp
45 * // Create agent with configuration
46 * libe3::E3Config config;
47 * config.link_layer = libe3::E3LinkLayer::ZMQ;
48 * config.transport_layer = libe3::E3TransportLayer::IPC;
49 * config.encoding = libe3::EncodingFormat::JSON;
50 * // Set custom ports if needed
51 * config.setup_port = 9990;
52 * config.subscriber_port = 9999;
53 * config.publisher_port = 9991;
54
55 * libe3::E3Agent agent(config);
56
57 * // Register Service Models
58 * agent.register_sm(std::make_unique<MySpectrumSM>());
59
60 * // Start the agent
61 * if (agent.start() != libe3::ErrorCode::SUCCESS) {
62 * // Handle error
63 * }
64
65 * // ... agent runs, handling dApp connections and messages ...
66
67 * // Shutdown
68 * agent.stop();
69 * ```
70 *
71 * **Thread Safety:**
72 * All public methods are thread-safe. The agent manages its own internal
73 * threads for protocol handling.
74 */
