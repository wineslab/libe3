41/**
42 * @brief E3Agent - Main façade for RAN vendor integration
43 *
44 * This class is the primary entry point for integrating E3AP functionality
45 * into a RAN function (DU, CU-CP, CU-UP). It provides:
46 *
47 * - Simple lifecycle management (init, start, stop, destroy)
48 * - Service Model registration
49 * - Automatic routing of control actions to the correct Service Model
50 * - Thread-safe operation
51 *
52 * **Usage Example:**
53 * ```cpp
54 * // Create agent with configuration
55 * libe3::E3Config config;
56 * config.link_layer = libe3::E3LinkLayer::ZMQ;
57 * config.transport_layer = libe3::E3TransportLayer::IPC;
58 * config.encoding = libe3::EncodingFormat::JSON;
59 * // Set custom ports if needed
60 * config.setup_port = 9990;
61 * config.subscriber_port = 9999;
62 * config.publisher_port = 9991;
63
64 * libe3::E3Agent agent(config);
65
66 * // Register Service Models
67 * agent.register_sm(std::make_unique<MySpectrumSM>());
68
69 * // Start the agent
70 * if (agent.start() != libe3::ErrorCode::SUCCESS) {
71 * // Handle error
72 * }
73
74 * // ... agent runs, handling dApp connections and messages ...
75
76 * // Shutdown
77 * agent.stop();
78 * ```
79 *
80 * **Thread Safety:**
81 * All public methods are thread-safe. The agent manages its own internal
82 * threads for protocol handling.
83 */
