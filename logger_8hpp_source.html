101 /**
102 * @brief Set log file path (append mode)
103 *
104 * If set, logs are written to this file instead of stderr
105 * when no callback is configured.
106 */
107 void set_log_file(const std::string& path) {
108 std::lock_guard<std::mutex> lock(mutex_);
109 if (file_) {
110 std::fclose(file_);
111 file_ = nullptr;
112 }
113
114 if (!path.empty()) {
115 file_ = std::fopen(path.c_str(), "a");
116 if (file_) {
117 std::setvbuf(file_, nullptr, _IOLBF, 0);
118 } else {
119 std::fprintf(stderr, "[Logger] Failed to open log file %s\n", path.c_str());
120 }
121 }
122 }
123
124 /**
125 * @brief Check if a level should be logged
126 */
127 bool should_log(LogLevel level) const noexcept {
128 return static_cast<int>(level) <= static_cast<int>(level_);
129 }
130
131 /**
132 * @brief Log a message
133 */
134 void log(LogLevel level, const std::string& component, const std::string& message) {
135 if (!should_log(level)) {
136 return;
137 }
138
139 std::lock_guard<std::mutex> lock(mutex_);
140
141 if (callback_) {
142 callback_(level, component, message);
143 } else {
144 FILE* out = file_ ? file_ : stderr;
145 // Default: write to stderr with timestamp
146 auto now = std::chrono::system_clock::now();
147 auto time_t = std::chrono::system_clock::to_time_t(now);
148 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
149 now.time_since_epoch()) % 1000;
150
151 char time_buf[32];
152 std::strftime(time_buf, sizeof(time_buf), "%Y-%m-%d %H:%M:%S", std::localtime(&time_t));
153
154 std::fprintf(out, "[%s.%03d] [%s] [%s] %s\n",
155 time_buf, static_cast<int>(ms.count()),
156 level_to_string(level), component.c_str(), message.c_str());
157 std::fflush(out);
158 }
159 }
160
161private:
162 Logger() = default;
163 ~Logger() = default;
164 Logger(const Logger&) = delete;
165 Logger& operator=(const Logger&) = delete;

171 case LogLevel::INFO: return "INFO ";
172 case LogLevel::DEBUG: return "DEBUG";
173 case LogLevel::TRACE: return "TRACE";
174 default: return "?????";
175 }
176 }
