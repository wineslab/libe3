44std::pair<ErrorCode, uint32_t> SubscriptionManager::register_dapp() {
45 std::unique_lock lock(mutex_);
46
47 // Find the next available dApp ID
48 uint32_t assigned_id = next_dapp_id_;
49 uint32_t attempts = 0;
50
51 // Search for an available ID (handle wrap-around and gaps from unregistered dApps)
52 while (registered_dapps_.count(assigned_id) > 0 && attempts <= MAX_DAPP_ID) {
53 assigned_id++;
54 if (assigned_id > MAX_DAPP_ID) {
55 assigned_id = MIN_DAPP_ID;
56 }
57 attempts++;
58 }
59
60 if (attempts > MAX_DAPP_ID) {
61 E3_LOG_ERROR(LOG_TAG) << "No available dApp IDs (range " << MIN_DAPP_ID
62 << ".." << MAX_DAPP_ID << ")";
63 return {ErrorCode::INTERNAL_ERROR, 0};
64 }
65
66 DAppEntry entry;
67 entry.dapp_identifier = assigned_id;
68 entry.registered_time = std::chrono::steady_clock::now();
69 registered_dapps_[assigned_id] = entry;
70
71 // Initialize empty subscription set for this dApp
72 dapp_subscriptions_[assigned_id] = {};
73
74 // Update next_dapp_id_ for the next registration
75 next_dapp_id_ = assigned_id + 1;
76 if (next_dapp_id_ > MAX_DAPP_ID) {
77 next_dapp_id_ = MIN_DAPP_ID;
78 }
79
80 E3_LOG_INFO(LOG_TAG) << "dApp registered successfully with ID " << assigned_id;
81 return {ErrorCode::SUCCESS, assigned_id};
82}

84ErrorCode SubscriptionManager::unregister_dapp(uint32_t dapp_id) {
85 std::unique_lock lock(mutex_);
86
87 auto it = registered_dapps_.find(dapp_id);
88 if (it == registered_dapps_.end()) {
89 E3_LOG_WARN(LOG_TAG) << "dApp " << dapp_id << " not found for unregistration";
90 return ErrorCode::DAPP_NOT_REGISTERED;
91 }
92
93 // Track affected RAN functions before removing subscriptions
94 std::vector<std::pair<uint32_t, bool>> affected_ran_functions;
95
96 auto sub_it = dapp_subscriptions_.find(dapp_id);
97 if (sub_it != dapp_subscriptions_.end()) {
98 for (uint32_t ran_func : sub_it->second) {
99 // Check if it currently has subscribers (before we remove this dApp)
100 auto& subscribers = ran_function_subscribers_[ran_func];
101 bool had_subscribers = !subscribers.empty();
102
103 // Remove this dApp from the RAN function's subscriber list
104 subscribers.erase(dapp_id);
105
106 // Track whether this RAN function still has subscribers
107 bool still_has_subscribers = !subscribers.empty();
108 if (had_subscribers && !still_has_subscribers) {
109 affected_ran_functions.emplace_back(ran_func, false); // should_start = false
110 }
111
112 // Remove subscription ID mapping and details
113 uint64_t sub_key = make_sub_key(dapp_id, ran_func);
114 auto sub_id_it = subscription_id_reverse_.find(sub_key);
115 if (sub_id_it != subscription_id_reverse_.end()) {
116 subscription_ids_.erase(sub_id_it->second);
117 subscription_id_reverse_.erase(sub_id_it);
118 }
119 subscription_details_.erase(sub_key);
120 }
121 dapp_subscriptions_.erase(sub_it);
122 }
123
124 // Remove the dApp registration
125 registered_dapps_.erase(it);
126
127 size_t subscriptions_removed = affected_ran_functions.size();
128 E3_LOG_INFO(LOG_TAG) << "dApp " << dapp_id << " unregistered, "
129 << subscriptions_removed << " subscriptions removed";
130
131 // Notify about SM lifecycle changes (outside the lock to avoid deadlock)
132 lock.unlock();
133 for (const auto& [ran_func, should_start] : affected_ran_functions) {
134 check_sm_lifecycle(ran_func, true); // had_subscribers = true
135 }
136
137 return ErrorCode::SUCCESS;
138}

159std::pair<ErrorCode, uint32_t> SubscriptionManager::add_subscription(
160 uint32_t dapp_id,
161 uint32_t ran_function_id,
162 const std::vector<uint32_t>& telemetry_ids,
163 const std::vector<uint32_t>& control_ids,
164 uint32_t periodicity_us) {
165 std::unique_lock lock(mutex_);
166
167 // Check if dApp is registered
168 if (registered_dapps_.count(dapp_id) == 0) {
169 E3_LOG_ERROR(LOG_TAG) << "dApp " << dapp_id << " not registered, cannot add subscription";
170 return {ErrorCode::DAPP_NOT_REGISTERED, 0};
171 }
172
173 // Check if subscription already exists
174 auto& dapp_subs = dapp_subscriptions_[dapp_id];
175 if (dapp_subs.count(ran_function_id) > 0) {
176 // Return existing subscription ID
177 uint64_t sub_key = make_sub_key(dapp_id, ran_function_id);
178 uint32_t existing_sub_id = subscription_id_reverse_[sub_key];
179 E3_LOG_WARN(LOG_TAG) << "dApp " << dapp_id << " already subscribed to RAN function "
180 << ran_function_id << " (subscription_id=" << existing_sub_id << ")";
181 return {ErrorCode::SUBSCRIPTION_EXISTS, existing_sub_id};
182 }
183
184 // Check if this is the first subscriber for this RAN function
185 bool had_subscribers = !ran_function_subscribers_[ran_function_id].empty();
186
187 // Assign subscription ID
188 uint32_t subscription_id = next_subscription_id_++;
189 if (next_subscription_id_ > 100) {
190 next_subscription_id_ = 1; // Wrap around per spec (0-100 range)
191 }
192
193 // Add subscription
194 dapp_subs.insert(ran_function_id);
195 ran_function_subscribers_[ran_function_id].insert(dapp_id);
196
197 // Add subscription ID mappings
198 subscription_ids_[subscription_id] = {dapp_id, ran_function_id};
199 uint64_t sub_key = make_sub_key(dapp_id, ran_function_id);
200 subscription_id_reverse_[sub_key] = subscription_id;
201
202 // Store per-subscription metadata
203 SubscriptionDetails details;
204 details.telemetry_ids = telemetry_ids;
205 details.control_ids = control_ids;
206 details.periodicity_us = periodicity_us;
207 subscription_details_[sub_key] = std::move(details);
208
209 E3_LOG_INFO(LOG_TAG) << "Subscription added: dApp " << dapp_id
210 << " -> RAN function " << ran_function_id
211 << " (subscription_id=" << subscription_id
212 << ", periodicity_us=" << periodicity_us
213 << ", telemetry_ids=" << telemetry_ids.size()
214 << ", control_ids=" << control_ids.size() << ")";
215
216 // Notify about SM lifecycle change if this is the first subscriber
217 lock.unlock();
218 if (!had_subscribers) {
219 check_sm_lifecycle(ran_function_id, false); // had_subscribers = false
220 }
221
222 return {ErrorCode::SUCCESS, subscription_id};
223}

225ErrorCode SubscriptionManager::remove_subscription(uint32_t dapp_id, uint32_t ran_function_id) {
226 std::unique_lock lock(mutex_);
227
228 auto dapp_it = dapp_subscriptions_.find(dapp_id);
229 if (dapp_it == dapp_subscriptions_.end()) {
230 return ErrorCode::SUBSCRIPTION_NOT_FOUND;
231 }
232
233 auto& dapp_subs = dapp_it->second;
234 if (dapp_subs.erase(ran_function_id) == 0) {
235 E3_LOG_WARN(LOG_TAG) << "Subscription not found for dApp " << dapp_id
236 << " -> RAN function " << ran_function_id;
237 return ErrorCode::SUBSCRIPTION_NOT_FOUND;
238 }
239
240 // Remove from reverse index
241 auto& subscribers = ran_function_subscribers_[ran_function_id];
242 bool had_subscribers = !subscribers.empty();
243 subscribers.erase(dapp_id);
244 bool still_has_subscribers = !subscribers.empty();
245
246 // Remove subscription ID mappings and details
247 uint64_t sub_key = make_sub_key(dapp_id, ran_function_id);
248 auto sub_id_it = subscription_id_reverse_.find(sub_key);
249 if (sub_id_it != subscription_id_reverse_.end()) {
250 subscription_ids_.erase(sub_id_it->second);
251 subscription_id_reverse_.erase(sub_id_it);
252 }
253 subscription_details_.erase(sub_key);
254
255 E3_LOG_INFO(LOG_TAG) << "Subscription removed: dApp " << dapp_id
256 << " -> RAN function " << ran_function_id;
257
258 // Notify about SM lifecycle change if this was the last subscriber
259 lock.unlock();
260 if (had_subscribers && !still_has_subscribers) {
261 check_sm_lifecycle(ran_function_id, true); // had_subscribers = true
262 }
263
264 return ErrorCode::SUCCESS;
265}

267ErrorCode SubscriptionManager::remove_subscription_by_id(uint32_t dapp_id, uint32_t subscription_id) {
268 std::unique_lock lock(mutex_);
269
270 // Find the subscription by ID
271 auto sub_it = subscription_ids_.find(subscription_id);
272 if (sub_it == subscription_ids_.end()) {
273 E3_LOG_WARN(LOG_TAG) << "Subscription ID " << subscription_id << " not found";
274 return ErrorCode::SUBSCRIPTION_NOT_FOUND;
275 }
276
277 auto [sub_dapp_id, ran_function_id] = sub_it->second;
278
279 // Verify the dApp ID matches
280 if (sub_dapp_id != dapp_id) {
281 E3_LOG_WARN(LOG_TAG) << "Subscription ID " << subscription_id
282 << " does not belong to dApp " << dapp_id;
283 return ErrorCode::SUBSCRIPTION_NOT_FOUND;
284 }
285
286 // Remove from dApp subscriptions
287 auto dapp_it = dapp_subscriptions_.find(dapp_id);
288 if (dapp_it != dapp_subscriptions_.end()) {
289 dapp_it->second.erase(ran_function_id);
290 }
291
292 // Remove from reverse index
293 auto& subscribers = ran_function_subscribers_[ran_function_id];
294 bool had_subscribers = !subscribers.empty();
295 subscribers.erase(dapp_id);
296 bool still_has_subscribers = !subscribers.empty();
297
298 // Remove subscription ID mappings and details
299 uint64_t sub_key = make_sub_key(dapp_id, ran_function_id);
300 subscription_id_reverse_.erase(sub_key);
301 subscription_ids_.erase(sub_it);
302 subscription_details_.erase(sub_key);
303
304 E3_LOG_INFO(LOG_TAG) << "Subscription " << subscription_id << " removed: dApp " << dapp_id
305 << " -> RAN function " << ran_function_id;
306
307 // Notify about SM lifecycle change if this was the last subscriber
308 lock.unlock();
309 if (had_subscribers && !still_has_subscribers) {
310 check_sm_lifecycle(ran_function_id, true); // had_subscribers = true
311 }
312
313 return ErrorCode::SUCCESS;
314}

424void SubscriptionManager::check_sm_lifecycle(uint32_t ran_function_id, bool had_subscribers) {
425 if (!sm_lifecycle_callback_) {
426 return;
427 }
428
429 // Determine if we need to start or stop
430 bool has_subscribers_now = has_subscribers(ran_function_id);
431
432 if (!had_subscribers && has_subscribers_now) {
433 // First subscriber - start SM
434 E3_LOG_INFO(LOG_TAG) << "Starting SM for RAN function " << ran_function_id
435 << " (first subscriber)";
436 sm_lifecycle_callback_(ran_function_id, true);
437 } else if (had_subscribers && !has_subscribers_now) {
438 // Last subscriber gone - stop SM
439 E3_LOG_INFO(LOG_TAG) << "Stopping SM for RAN function " << ran_function_id
440 << " (no remaining subscribers)";
441 sm_lifecycle_callback_(ran_function_id, false);
442 }
