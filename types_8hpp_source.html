45/**
46 * @brief Role this E3 endpoint plays in the protocol.
47 *
48 * RAN is the existing behaviour: the agent binds the setup/inbound/outbound
49 * sockets and answers dApp requests. DAPP is the symmetric client side: the
50 * agent connects to those endpoints and drives setup/subscribe/release.
51 *
52 * Default is RAN so that any caller compiled against earlier libe3 headers
53 * keeps today's behaviour.
54 */
55enum class E3Role : uint8_t {
56 RAN = 0, ///< RAN function (binds sockets; today's only behaviour)
57 DAPP = 1 ///< dApp (connects to a RAN agent)
58};

309/**
310 * @brief Generic E3AP PDU structure
311 */
312struct Pdu {
313 PduType type{PduType::SETUP_REQUEST};
314 PduChoice choice;
315 uint32_t message_id{0}; ///< Unique message identifier
316 uint64_t timestamp{0}; ///< Message timestamp (milliseconds since epoch)
317
318 Pdu() : timestamp(static_cast<uint64_t>(
319 std::chrono::duration_cast<std::chrono::milliseconds>(
320 std::chrono::system_clock::now().time_since_epoch()
321 ).count())) {}
322
323 explicit Pdu(PduType t) : type(t), timestamp(static_cast<uint64_t>(
324 std::chrono::duration_cast<std::chrono::milliseconds>(
325 std::chrono::system_clock::now().time_since_epoch()
326 ).count())) {}
327
328 template<typename T>
329 Pdu(PduType t, T&& data) : type(t), choice(std::forward<T>(data)),
330 timestamp(static_cast<uint64_t>(
331 std::chrono::duration_cast<std::chrono::milliseconds>(
332 std::chrono::system_clock::now().time_since_epoch()
333 ).count())) {}
334
335 /**
336 * @brief Get PDU data with type checking
337 */
338 template<typename T>
339 T* get_if() noexcept {
340 return std::get_if<T>(&choice);
341 }
342
343 template<typename T>
344 const T* get_if() const noexcept {
345 return std::get_if<T>(&choice);

368 E3TransportLayer transport_layer{E3TransportLayer::IPC}; ///< Transport-layer protocol (SCTP, TCP, IPC)

378 std::string publisher_endpoint{"ipc:///tmp/dapps/e3_socket"}; ///< Endpoint URI for outbound channel

383#elif defined(LIBE3_ENABLE_JSON)

385#else // Fallback
386 EncodingFormat encoding{EncodingFormat::ASN1}; ///< PDU encoding format (ASN.1 PER or JSON)
387#endif

391 uint32_t recv_timeout_ms{1000}; ///< Receive operation timeout in milliseconds
392 uint32_t send_timeout_ms{1000}; ///< Send operation timeout in milliseconds
393
394 // Buffer sizes
395 size_t receive_buffer_size{DEFAULT_BUFFER_SIZE}; ///< Size of the receive buffer in bytes
396 size_t send_buffer_size{DEFAULT_BUFFER_SIZE}; ///< Size of the send buffer in bytes
397
398 // Threading
399 size_t io_threads{2}; ///< Number of I/O threads (used by ZMQ context)
400
401 /**
402 * @brief CPU core to pin the subscriber and publisher I/O threads to.
403 *
404 * When set to a non-negative value the subscriber_loop and publisher_loop
405 * threads will be pinned to this logical CPU core via
406 * pthread_setaffinity_np (Linux only; ignored on other platforms).
407 * This eliminates cache-migration overhead and reduces scheduling jitter
408 * in sub-millisecond control loops.
409 *
410 * -1 (default) disables core pinning.
411 */
412 int io_thread_affinity{-1};
413
414 /**
415 * @brief Nice value applied to the subscriber and publisher I/O threads.
416 *
417 * Range -20 (highest priority) to 19 (lowest priority), 0 = system
418 * default. A negative value reduces OS scheduling latency at the cost of
419 * slightly less CPU time for other processes. Requires appropriate
420 * privileges for negative values (CAP_SYS_NICE on Linux).
421 *
422 * Applied via setpriority(PRIO_PROCESS, tid, value) on Linux.
423 */
424 int io_thread_niceness{0};
425
426 // Logging level (0=none, 1=error, 2=warn, 3=info, 4=debug, 5=trace)
427 int log_level{3}; ///< Logging verbosity: 0=none, 1=error, 2=warn, 3=info, 4=debug, 5=trace
428
429 /**
430 * @brief Path of the shared log file. Empty (default) => auto-generated.
431 *
432 * When left empty the library derives a per-process path of the form
433 * `/tmp/e3_<role>[_<sanitized-id>]_<euid>.log` (see E3Interface ctor).
434 * The trailing effective-uid keeps the file owned by whoever opens it,
435 * which avoids `fs.protected_regular` blocking an O_CREAT re-open of a
436 * /tmp file owned by a different user (e.g. a root gNB vs. a prior
437 * non-sudo run). Set this explicitly to override the location entirely.
438 */
439 std::string log_path;
